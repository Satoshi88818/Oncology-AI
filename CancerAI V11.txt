Fix: max_mb now enforced via streaming read before file.read() is called.

""" Input validators — v11. FIX: validate_upload_file now enforces the max_mb limit by reading the upload stream in chunks, rejecting the request before the full file is buffered in memory. This prevents a 10 GB SVS file from causing an OOM. """ from __future__ import annotations import json from fastapi import HTTPException, UploadFile, status async def validate_upload_file( file: UploadFile, allowed_types: set[str], max_mb: float = 500, ) -> bytes: """ Validate content type and size, returning the raw file bytes. Reads the upload stream in 1 MB chunks and raises HTTP 413 if the total exceeds max_mb before completing the read. Returns: The complete file bytes (so the caller does not need to re-read). """ if file.content_type not in allowed_types: raise HTTPException( status_code=status.HTTP_415_UNSUPPORTED_MEDIA_TYPE, detail=f"Unsupported file type: {file.content_type}. Allowed: {allowed_types}", ) max_bytes = int(max_mb * 1024 * 1024) chunk_size = 1024 * 1024 # 1 MB chunks total = 0 chunks: list[bytes] = [] while True: chunk = await file.read(chunk_size) if not chunk: break total += len(chunk) if total > max_bytes: raise HTTPException( status_code=status.HTTP_413_REQUEST_ENTITY_TOO_LARGE, detail=f"File exceeds maximum allowed size of {max_mb} MB.", ) chunks.append(chunk) return b"".join(chunks) def validate_genomic_payload(raw: str) -> dict: """ Parse and validate genomic payload JSON. Validates: - Valid JSON - 'features' key present and is a non-empty list of finite numbers - No NaN strings masquerading as floats - ECOG status in [0, 4] if present """ try: payload = json.loads(raw) except (json.JSONDecodeError, TypeError): raise HTTPException( status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail="Invalid JSON in genomic payload.", ) features = payload.get("features") if features is None: raise HTTPException( status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail="'features' key missing from genomic payload.", ) if not isinstance(features, list) or len(features) == 0: raise HTTPException( status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail="'features' must be a non-empty list.", ) for i, v in enumerate(features): if not isinstance(v, (int, float)): raise HTTPException( status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail=f"Non-numeric value at features[{i}]: {v!r}", ) ecog = payload.get("ecog_status") if ecog is not None and not (0 <= int(ecog) <= 4): raise HTTPException( status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail=f"ecog_status must be in [0, 4], got {ecog}.", ) return payload 

app/utils/audit.py

Fix: key-versioned Fernet encryption so past audit entries survive key rotation.

""" Key-versioned encrypted audit logger — v11. FIX: AuditLogger now embeds a key_version identifier alongside each encrypted entry so that AuditExporter can select the correct Fernet key when decrypting historical records after a key rotation. Usage: # Logging (in predictor / endpoints) audit_logger = AuditLogger() audit_logger.log(user="alice", action="predict_image", cancer_prob=0.73) # Decryption / export (authorised admin only) from app.utils.audit import AuditExporter exporter = AuditExporter(key_store={"v1": "<fernet_key_v1>", "v2": "<fernet_key_v2>"}) records = exporter.decrypt_log_file("logs/audit.log") """ from __future__ import annotations import json import logging import os from datetime import datetime, timezone from cryptography.fernet import Fernet, InvalidToken logger = logging.getLogger("cancerAI.audit") def _load_key_store() -> dict[str, bytes]: """ Build a key store from environment variables. Convention: ENCRYPTION_KEY_V1=<base64-fernet-key> ← oldest key ENCRYPTION_KEY_V2=<base64-fernet-key> ← current key ENCRYPTION_KEY_CURRENT=v2 ← which version to use for new entries At minimum, ENCRYPTION_KEY_V1 and ENCRYPTION_KEY_CURRENT=v1 must be set. """ store: dict[str, bytes] = {} for key, value in os.environ.items(): if key.startswith("ENCRYPTION_KEY_V") and value: version_tag = key.removeprefix("ENCRYPTION_KEY_") store[version_tag] = value.encode() if isinstance(value, str) else value if not store: # Fallback: accept legacy ENCRYPTION_KEY (treated as V1) legacy = os.getenv("ENCRYPTION_KEY") if legacy: store["V1"] = legacy.encode() if isinstance(legacy, str) else legacy if not store: raise RuntimeError( "No Fernet encryption key found. Set ENCRYPTION_KEY_V1 (and ENCRYPTION_KEY_CURRENT=V1) " "in your environment, or generate one with:\n" " python -c \"from cryptography.fernet import Fernet; print(Fernet.generate_key().decode())\"" ) return store def _current_version() -> str: return os.getenv("ENCRYPTION_KEY_CURRENT", "V1") class AuditLogger: """Encrypts and emits audit log entries tagged with the active key version.""" def log(self, **fields) -> None: entry = { "timestamp": datetime.now(timezone.utc).isoformat(), **{k: str(v) for k, v in fields.items()}, } key_store = _load_key_store() current_version = _current_version() key = key_store.get(current_version) if key is None: raise RuntimeError( f"ENCRYPTION_KEY_CURRENT='{current_version}' but no matching " f"ENCRYPTION_KEY_{current_version} found in environment." ) fernet = Fernet(key) ciphertext = fernet.encrypt(json.dumps(entry).encode()).decode() # Format: KEY_VERSION:CIPHERTEXT — allows AuditExporter to pick the right key logger.info("AUDIT:%s:%s", current_version, ciphertext) class AuditExporter: """ Decrypts audit log entries for authorised export. Args: key_store: mapping of version tag → Fernet key bytes, e.g. {"V1": b"<key>", "V2": b"<key>"}. If None, loads from environment variables. """ def __init__(self, key_store: dict[str, bytes] | None = None): self._store = key_store or _load_key_store() self._fernets = {v: Fernet(k) for v, k in self._store.items()} def decrypt_entry(self, raw_log_line: str) -> dict | None: """ Decrypt a single log line of the form 'AUDIT:VERSION:CIPHERTEXT'. Returns the decoded dict, or None if the line cannot be parsed/decrypted. """ if not raw_log_line.strip().startswith("AUDIT:"): return None parts = raw_log_line.strip().split(":", 2) # Support legacy format 'AUDIT:CIPHERTEXT' (no version tag) if len(parts) == 2: version, ciphertext = "V1", parts[1] elif len(parts) == 3: _, version, ciphertext = parts else: return None fernet = self._fernets.get(version) if fernet is None: logger.warning("No decryption key for version '%s' — skipping entry.", version) return None try: return json.loads(fernet.decrypt(ciphertext.encode()).decode()) except (InvalidToken, Exception) as exc: logger.error("Failed to decrypt audit entry (version=%s): %s", version, exc) return None def decrypt_log_file(self, log_path: str) -> list[dict]: """Decrypt all AUDIT entries from a log file. Returns list of decoded records.""" records: list[dict] = [] with open(log_path) as f: for line in f: record = self.decrypt_entry(line) if record is not None: records.append(record) return records
