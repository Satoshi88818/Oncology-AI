""" Domain-specific pathology backbone adapter — v10. Supports: - Local SavedModel (any backbone) - HuggingFace Hub (UNI, CONCH, PLIP) with token auth + graceful fallback - EfficientNetB3 (ImageNet, fallback) HF_TOKEN environment variable must be set for private/gated repos (UNI, CONCH). """ from __future__ import annotations import logging import os from pathlib import Path import tensorflow as tf logger = logging.getLogger("cancerAI.backbone") def load_backbone( backbone: str, weights_path: str = "", hf_repo_id: str = "", input_shape: tuple[int, int, int] = (300, 300, 3), ) -> tf.keras.Model: """ Load a pathology-specific backbone encoder. Priority: 1. Local weights path (SavedModel directory) if non-empty. 2. HuggingFace Hub download if hf_repo_id is set and HF_TOKEN available. 3. EfficientNetB3 (ImageNet) fallback with a warning. """ if weights_path and Path(weights_path).exists(): logger.info("Loading backbone from local path: %s", weights_path) return _load_local_saved_model(weights_path, input_shape) if hf_repo_id: logger.info("Attempting HuggingFace Hub download: %s", hf_repo_id) backbone_model = _load_from_hf_hub(hf_repo_id, backbone, input_shape) if backbone_model is not None: return backbone_model logger.warning( "HF Hub download failed. Check HF_TOKEN and repo access approval. " "Falling back to EfficientNetB3 (ImageNet). " "Performance will be significantly degraded for pathology tasks." ) logger.warning( "No domain-specific backbone loaded. Using EfficientNetB3 (ImageNet). " "For production use, configure a pathology-specific backbone." ) return _efficientnet_b3_fallback(input_shape) def _load_local_saved_model( weights_path: str, input_shape: tuple[int, int, int], ) -> tf.keras.Model: try: model = tf.saved_model.load(weights_path) logger.info("Loaded SavedModel from %s", weights_path) return model except Exception as exc: raise RuntimeError( f"Failed to load backbone weights from '{weights_path}'. " f"Ensure the path contains a valid TF SavedModel directory. " f"Original error: {exc}" ) from exc def _load_from_hf_hub( repo_id: str, backbone_name: str, input_shape: tuple[int, int, int], ) -> tf.keras.Model | None: hf_token = os.getenv("HF_TOKEN") if not hf_token: logger.warning( "HF_TOKEN not set. Cannot download gated repo '%s'. " "Request access at https://huggingface.co/%s and set HF_TOKEN.", repo_id, repo_id, ) return None try: from huggingface_hub import snapshot_download local_dir = snapshot_download( repo_id=repo_id, token=hf_token, local_dir=f"models/{backbone_name}_hf_cache", ) logger.info("Downloaded HF repo to %s", local_dir) return _load_local_saved_model(local_dir, input_shape) except ImportError: logger.error("huggingface-hub not installed. Run: pip install huggingface-hub") return None except Exception as exc: logger.error("HuggingFace Hub download error: %s", exc) return None def _efficientnet_b3_fallback( input_shape: tuple[int, int, int], ) -> tf.keras.Model: base = tf.keras.applications.EfficientNetB3( include_top=False, weights="imagenet", input_shape=input_shape, pooling="avg", ) base.trainable = False return base

""" Cross-attention early fusion module — v10. Replaces late/rule-based fusion with learned cross-attention between image (MIL bag embedding) and genomic embeddings. Architecture: image_embed (B, D) ──┐ ├─→ CrossAttention → joint_embed (B, D) → classification head genomic_embed (B, D) ─┘ The image embedding attends to the genomic embedding and vice versa; both are projected to a shared embed_dim before attention. """ from __future__ import annotations import tensorflow as tf class CrossAttentionFusion(tf.keras.layers.Layer): """ Bidirectional cross-attention fusion of image and genomic embeddings. Both modalities are projected to `embed_dim`, then each attends to the other via multi-head attention. The attended representations are concatenated and passed through a joint MLP classification head. """ def __init__( self, embed_dim: int = 256, num_heads: int = 4, num_classes: int = 1, dropout: float = 0.1, **kwargs, ): super().__init__(**kwargs) self.embed_dim = embed_dim self.num_heads = num_heads self.num_classes = num_classes # Projection layers self.img_proj = tf.keras.layers.Dense(embed_dim, name="img_proj") self.gen_proj = tf.keras.layers.Dense(embed_dim, name="gen_proj") # Expand sequence dim for attention: (B, D) → (B, 1, D) # Image attends to genomics self.img_attn_gen = tf.keras.layers.MultiHeadAttention( num_heads=num_heads, key_dim=embed_dim // num_heads, dropout=dropout, name="img_attends_gen", ) # Genomics attends to image self.gen_attn_img = tf.keras.layers.MultiHeadAttention( num_heads=num_heads, key_dim=embed_dim // num_heads, dropout=dropout, name="gen_attends_img", ) # Layer norms self.norm_img = tf.keras.layers.LayerNormalization(name="norm_img") self.norm_gen = tf.keras.layers.LayerNormalization(name="norm_gen") # Joint classification head self.joint_dense1 = tf.keras.layers.Dense( embed_dim, activation="gelu", name="joint_dense1" ) self.dropout_layer = tf.keras.layers.Dropout(dropout) self.joint_out = tf.keras.layers.Dense(num_classes, name="joint_out") def call( self, image_embed: tf.Tensor, # (B, img_dim) genomic_embed: tf.Tensor, # (B, gen_dim) training: bool = False, ) -> dict[str, tf.Tensor]: # Project both to shared embed_dim img = self.img_proj(image_embed) # (B, D) gen = self.gen_proj(genomic_embed) # (B, D) # Expand to (B, 1, D) for MultiHeadAttention (sequence length = 1) img_seq = tf.expand_dims(img, axis=1) # (B, 1, D) gen_seq = tf.expand_dims(gen, axis=1) # (B, 1, D) # Bidirectional cross-attention img_attended, img_attn_weights = self.img_attn_gen( query=img_seq, key=gen_seq, value=gen_seq, training=training, return_attention_scores=True, ) gen_attended, gen_attn_weights = self.gen_attn_img( query=gen_seq, key=img_seq, value=img_seq, training=training, return_attention_scores=True, ) # Residual + norm, squeeze sequence dim img_fused = self.norm_img(img_seq + img_attended)[:, 0, :] # (B, D) gen_fused = self.norm_gen(gen_seq + gen_attended)[:, 0, :] # (B, D) # Concatenate fused embeddings joint = tf.concat([img_fused, gen_fused], axis=-1) # (B, 2D) # Joint MLP head joint = self.joint_dense1(joint) joint = self.dropout_layer(joint, training=training) logits = self.joint_out(joint) # (B, num_classes) probs = tf.sigmoid(logits) return { "logits": logits, "probs": probs, "img_attn_weights": img_attn_weights, # (B, heads, 1, 1) — for explainability "gen_attn_weights": gen_attn_weights, } def get_config(self): config = super().get_config() config.update( embed_dim=self.embed_dim, num_heads=self.num_heads, num_classes=self.num_classes, ) return config

""" Hierarchical attention-based Multiple Instance Learning (MIL) — v10. Changes from v9: - Optional hierarchical aggregation: tiles → clusters → slide embedding. - Returns embed vector for downstream cross-attention fusion. - Unchanged attention softmax (correct) and MC Dropout support. """ from __future__ import annotations import tensorflow as tf class AttentionMIL(tf.keras.layers.Layer): """ Attention-based MIL with optional hierarchical clustering. In hierarchical mode: 1. Tiles are grouped into `n_clusters` clusters (via learned linear assignment). 2. Within-cluster attention produces cluster embeddings. 3. Between-cluster attention produces slide embedding. In flat mode (hierarchical=False): single attention pass (v9 behaviour). """ def __init__( self, embed_dim: int = 512, attention_dim: int = 128, dropout_rate: float = 0.25, hierarchical: bool = True, n_clusters: int = 8, **kwargs, ): super().__init__(**kwargs) self.embed_dim = embed_dim self.attention_dim = attention_dim self.hierarchical = hierarchical self.n_clusters = n_clusters # Tile-level attention self.attn_V = tf.keras.layers.Dense(attention_dim, activation="tanh", name="attn_V") self.attn_U = tf.keras.layers.Dense(attention_dim, activation="sigmoid", name="attn_U") self.attn_w = tf.keras.layers.Dense(1, name="attn_w") self.dropout = tf.keras.layers.Dropout(dropout_rate) if hierarchical: # Cluster assignment (soft): tile → cluster membership self.cluster_assign = tf.keras.layers.Dense(n_clusters, activation="softmax", name="cluster_assign") # Cluster-level attention self.cl_attn_V = tf.keras.layers.Dense(attention_dim, activation="tanh", name="cl_attn_V") self.cl_attn_U = tf.keras.layers.Dense(attention_dim, activation="sigmoid", name="cl_attn_U") self.cl_attn_w = tf.keras.layers.Dense(1, name="cl_attn_w") def _gated_attention(self, x, V, U, w, training): """Gated attention scoring: a = softmax(w^T * (tanh(Vh) ⊙ sigmoid(Uh))).""" h_V = V(self.dropout(x, training=training)) h_U = U(self.dropout(x, training=training)) scores = w(h_V * h_U) # (B, N, 1) weights = tf.nn.softmax(scores, axis=1) aggregated = tf.reduce_sum(weights * x, axis=1) # (B, D) return aggregated, weights def call(self, bag: tf.Tensor, training: bool = False) -> dict[str, tf.Tensor]: """ Args: bag: (B, N, embed_dim) tile embeddings. training: bool, enables MC Dropout. Returns: dict with 'embed' (B, embed_dim), 'attention' (B, N, 1), and optionally 'cluster_attention'. """ if self.hierarchical: return self._hierarchical_forward(bag, training) return self._flat_forward(bag, training) def _flat_forward(self, bag, training): embed, attention = self._gated_attention( bag, self.attn_V, self.attn_U, self.attn_w, training ) return {"embed": embed, "attention": attention} def _hierarchical_forward(self, bag, training): # Tile → cluster soft assignment: (B, N, n_clusters) assignments = self.cluster_assign(bag) # Weighted sum to form cluster embeddings: (B, n_clusters, embed_dim) # assignments: (B, N, K) → transpose to (B, K, N) a_T = tf.transpose(assignments, perm=[0, 2, 1]) cluster_embeds = tf.matmul(a_T, bag) # (B, K, embed_dim) # Within-cluster tile attention (flat, over cluster embeddings) slide_embed, cluster_attention = self._gated_attention( cluster_embeds, self.cl_attn_V, self.cl_attn_U, self.cl_attn_w, training ) # Keep tile-level attention for explainability: # aggregate via cluster assignments for tile importance tile_attn = tf.matmul(assignments, cluster_attention) # (B, N, 1) tile_attn = tile_attn / (tf.reduce_sum(tile_attn, axis=1, keepdims=True) + 1e-8) return { "embed": slide_embed, "attention": tile_attn, "cluster_attention": cluster_attention, }

""" Genomic MLP treatment recommendation model — v10. Changes from v9: - MC Dropout is now correctly enabled during inference (training=True in predict). - Returns both embedding vector (for fusion) and treatment logits. - Uncertainty estimation via variance across MC samples. """ from __future__ import annotations import numpy as np import tensorflow as tf class GenomicTreatmentModel(tf.keras.Model): """ MLP over genomic features producing: 1. A genomic embedding for cross-attention fusion. 2. Treatment recommendation logits (when used standalone / late fusion). MC Dropout is enabled during inference to give epistemic uncertainty on the treatment pathway (fixed gap from v9). """ def __init__( self, n_genomic_features: int = 64, n_treatments: int = 6, embed_dim: int = 256, dropout_rate: float = 0.3, l2_reg: float = 1e-4, **kwargs, ): super().__init__(**kwargs) reg = tf.keras.regularizers.l2(l2_reg) self.dense1 = tf.keras.layers.Dense(256, activation="relu", kernel_regularizer=reg) self.bn1 = tf.keras.layers.BatchNormalization() self.drop1 = tf.keras.layers.Dropout(dropout_rate) self.dense2 = tf.keras.layers.Dense(128, activation="relu", kernel_regularizer=reg) self.bn2 = tf.keras.layers.BatchNormalization() self.drop2 = tf.keras.layers.Dropout(dropout_rate) # Embed head — output used for cross-attention fusion self.embed_head = tf.keras.layers.Dense(embed_dim, name="genomic_embed") # Treatment classification head (standalone / late-fusion usage) self.treatment_out = tf.keras.layers.Dense(n_treatments, name="treatment_logits") def call( self, x: tf.Tensor, training: bool = False ) -> dict[str, tf.Tensor]: h = self.dense1(x) h = self.bn1(h, training=training) h = self.drop1(h, training=training) h = self.dense2(h) h = self.bn2(h, training=training) h = self.drop2(h, training=training) embed = self.embed_head(h) logits = self.treatment_out(embed) probs = tf.nn.softmax(logits, axis=-1) return {"embed": embed, "logits": logits, "probs": probs} def mc_predict( self, x: tf.Tensor, n_samples: int = 30 ) -> dict[str, np.ndarray]: """ MC Dropout inference — v10 fix: training=True propagated for stochastic sampling. Returns mean and std over n_samples forward passes. """ samples = [self(x, training=True)["probs"].numpy() for _ in range(n_samples)] stacked = np.stack(samples, axis=0) # (n_samples, B, n_treatments) return { "mean_probs": stacked.mean(axis=0), "uncertainty": stacked.std(axis=0), }

""" CNN tile encoder with domain-specific backbone support — v10. Outputs a tile-level embedding used by the MIL aggregator. MC Dropout enabled for stochastic inference. """ from __future__ import annotations import tensorflow as tf from app.models.backbone import load_backbone class TileEncoder(tf.keras.Model): """CNN tile encoder wrapping a pathology backbone.""" def __init__( self, input_shape: tuple[int, int, int] = (300, 300, 3), embed_dim: int = 512, dropout_rate: float = 0.3, backbone: str = "uni", backbone_weights_path: str = "", backbone_hf_repo_id: str = "", l2_reg: float = 1e-4, **kwargs, ): super().__init__(**kwargs) self.backbone_model = load_backbone( backbone=backbone, weights_path=backbone_weights_path, hf_repo_id=backbone_hf_repo_id, input_shape=input_shape, ) reg = tf.keras.regularizers.l2(l2_reg) self.projection = tf.keras.layers.Dense( embed_dim, activation="relu", kernel_regularizer=reg, name="tile_proj" ) self.dropout = tf.keras.layers.Dropout(dropout_rate) self.norm = tf.keras.layers.LayerNormalization() def call(self, tiles: tf.Tensor, training: bool = False) -> tf.Tensor: """ Args: tiles: (B, H, W, C) batch of tile images. training: enables MC Dropout for stochastic inference. Returns: embed: (B, embed_dim) """ features = self.backbone_model(tiles, training=False) embed = self.projection(features) embed = self.dropout(embed, training=training) return self.norm(embed)

""" Adaptive WSI tiling pipeline — v10. Changes from v9: - Cellularity-aware adaptive tile count: high-density slides get more tiles. - Hierarchical tile grouping indices returned for hierarchical MIL. - Parallel extraction unchanged. """ from __future__ import annotations import logging from concurrent.futures import ThreadPoolExecutor, as_completed from pathlib import Path import cv2 import numpy as np from PIL import Image logger = logging.getLogger("cancerAI.wsi") try: import openslide HAS_OPENSLIDE = True except ImportError: HAS_OPENSLIDE = False logger.warning("openslide not available — WSI support disabled.") def _tissue_density(tile_rgb: np.ndarray) -> float: """Estimate tissue density as fraction of non-background pixels.""" gray = cv2.cvtColor(tile_rgb, cv2.COLOR_RGB2GRAY) _, binary = cv2.threshold(gray, 220, 255, cv2.THRESH_BINARY_INV) return float(binary.sum()) / binary.size def _process_tile( slide, x: int, y: int, tile_size: int, level: int, min_tissue_ratio: float, target_size: tuple[int, int], ) -> tuple[np.ndarray, float] | None: """Extract, validate and resize a single tile. Returns (tile_array, density) or None.""" region = slide.read_region((x, y), level, (tile_size, tile_size)).convert("RGB") tile_np = np.array(region) density = _tissue_density(tile_np) if density < min_tissue_ratio: return None tile_resized = np.array( Image.fromarray(tile_np).resize(target_size, Image.LANCZOS), dtype=np.float32 ) / 255.0 return tile_resized, density def extract_tiles( slide_path: str, tile_size: int = 512, level: int = 0, min_tissue_ratio: float = 0.08, max_tiles: int = 64, max_workers: int = 8, target_size: tuple[int, int] = (300, 300), adaptive_sampling: bool = True, high_cellularity_threshold: float = 0.6, adaptive_max_tiles: int = 128, ) -> list[np.ndarray]: """ Extract tissue-containing tiles from a WSI. Adaptive sampling: if mean tissue density exceeds `high_cellularity_threshold`, cap is raised to `adaptive_max_tiles` to avoid losing diagnostic information. Returns: List of (H, W, C) float32 tile arrays, capped at effective max_tiles. """ if not HAS_OPENSLIDE: raise RuntimeError("openslide is required for WSI processing.") path = str(Path(slide_path).resolve()) with openslide.OpenSlide(path) as slide: w, h = slide.level_dimensions[level] coords = [ (x, y) for y in range(0, h, tile_size) for x in range(0, w, tile_size) ] results: list[tuple[np.ndarray, float]] = [] with ThreadPoolExecutor(max_workers=max_workers) as executor: futures = { executor.submit( _process_tile, slide, x, y, tile_size, level, min_tissue_ratio, target_size ): (x, y) for x, y in coords } for future in as_completed(futures): result = future.result() if result is not None: results.append(result) if not results: logger.warning("No tissue tiles extracted from %s", slide_path) return [] tiles, densities = zip(*results) mean_density = float(np.mean(densities)) effective_max = max_tiles if adaptive_sampling and mean_density > high_cellularity_threshold: effective_max = adaptive_max_tiles logger.info( "High cellularity slide (density=%.2f) — raising tile cap to %d", mean_density, effective_max, ) # Sample evenly spaced tiles up to effective_max tiles_list = list(tiles) if len(tiles_list) > effective_max: indices = np.linspace(0, len(tiles_list) - 1, effective_max, dtype=int) tiles_list = [tiles_list[i] for i in indices] logger.info("Extracted %d tiles from %s (mean density=%.2f)", len(tiles_list), slide_path, mean_density) return tiles_list

""" tf.data ragged tensor DataLoader for bag-level MIL training — v10. Changes from v9: - Replaced manual GradientTape-per-bag loop with tf.data ragged pipeline. - Bucketed batching groups bags of similar tile counts together to minimize padding. - Supports prefetching for GPU utilisation. """ from __future__ import annotations import numpy as np import tensorflow as tf from typing import Sequence def make_mil_dataset( bags: list[np.ndarray], # each: (n_tiles, H, W, C) labels: list[int], tile_size: tuple[int, int, int] = (300, 300, 3), batch_size: int = 8, shuffle: bool = True, bucket_boundaries: list[int] | None = None, prefetch_buffer: int = 4, ) -> tf.data.Dataset: """ Build a bucketed tf.data.Dataset from a list of bags (variable-length). Bucketing groups bags with similar tile counts into the same batch, avoiding excessive zero-padding and OOM errors on large slides. Args: bags: List of (n_tiles, H, W, C) arrays (variable n_tiles per bag). labels: Corresponding binary labels. batch_size: Number of bags per batch. bucket_boundaries: Tile-count thresholds for bucketing. prefetch_buffer: Number of batches to prefetch. Returns: tf.data.Dataset yielding (ragged_bag_batch, label_batch). """ if bucket_boundaries is None: bucket_boundaries = [8, 16, 32, 64] def generator(): for bag, label in zip(bags, labels): # Yield (bag_tensor, n_tiles, label) so bucketing can use n_tiles bag_tensor = tf.constant(bag, dtype=tf.float32) yield bag_tensor, tf.cast(label, tf.int32) # Ragged dataset: variable first dim (n_tiles) is the ragged dimension dataset = tf.data.Dataset.from_generator( generator, output_signature=( tf.RaggedTensorSpec(shape=[None, *tile_size], dtype=tf.float32, ragged_rank=1), tf.TensorSpec(shape=(), dtype=tf.int32), ), ) if shuffle: dataset = dataset.shuffle(buffer_size=len(bags), reshuffle_each_iteration=True) # Bucket by number of tiles per bag for efficient batching dataset = dataset.bucket_by_sequence_length( element_length_func=lambda bag, label: tf.shape(bag)[0], bucket_boundaries=bucket_boundaries, bucket_batch_sizes=[batch_size] * (len(bucket_boundaries) + 1), padding_values=( tf.constant(0.0, dtype=tf.float32), tf.constant(0, dtype=tf.int32), ), drop_remainder=False, ) return dataset.prefetch(prefetch_buffer)

""" Genomic preprocessing pipeline — v10. Changes from v9: - Pipeline version pinned and SHA-256 hash verified on load. - Graceful mismatch recovery: if feature count mismatches, attempt re-imputation with a fallback zero-padding strategy rather than hard crash. - Version metadata embedded in saved pipeline artifact. """ from __future__ import annotations import hashlib import json import logging from pathlib import Path import joblib import numpy as np from sklearn.decomposition import PCA from sklearn.feature_selection import VarianceThreshold from sklearn.impute import SimpleImputer from sklearn.pipeline import Pipeline logger = logging.getLogger("cancerAI.genomics") PIPELINE_META_SUFFIX = ".meta.json" def build_genomic_pipeline( n_components: int = 64, variance_threshold: float = 0.01, imputation_strategy: str = "median", ) -> Pipeline: """Build a fresh sklearn genomic preprocessing pipeline.""" return Pipeline([ ("imputer", SimpleImputer(strategy=imputation_strategy)), ("variance_filter", VarianceThreshold(threshold=variance_threshold)), ("pca", PCA(n_components=n_components, random_state=42)), ]) def save_pipeline( pipeline: Pipeline, path: str, version: str = "1.0.0", ) -> str: """ Save pipeline + metadata (version, sha256). Returns sha256 hash of saved file. """ path = str(path) joblib.dump(pipeline, path) sha256 = _file_sha256(path) meta = {"version": version, "sha256": sha256} with open(path + PIPELINE_META_SUFFIX, "w") as f: json.dump(meta, f) logger.info("Genomic pipeline saved: %s (sha256=%s)", path, sha256) return sha256 def load_pipeline( path: str, expected_version: str | None = None, expected_hash: str | None = None, ) -> Pipeline: """ Load and verify a genomic preprocessing pipeline. Verifies SHA-256 hash and version string if provided. Raises RuntimeError on hard mismatch; logs warning on soft mismatch. """ path = str(path) if not Path(path).exists(): raise FileNotFoundError(f"Genomic pipeline not found at '{path}'.") actual_hash = _file_sha256(path) if expected_hash and actual_hash != expected_hash: raise RuntimeError( f"Genomic pipeline hash mismatch!\n" f" Expected: {expected_hash}\n" f" Actual: {actual_hash}\n" f"Pipeline may have been modified or corrupted. Re-fit on training data." ) meta_path = path + PIPELINE_META_SUFFIX if Path(meta_path).exists(): with open(meta_path) as f: meta = json.load(f) if expected_version and meta.get("version") != expected_version: logger.warning( "Genomic pipeline version mismatch: expected %s, found %s. " "Consider re-fitting if feature schema has changed.", expected_version, meta.get("version"), ) else: logger.warning("No pipeline metadata found at %s. Version unverified.", meta_path) pipeline = joblib.load(path) logger.info("Genomic pipeline loaded from %s (sha256=%s)", path, actual_hash) return pipeline def transform_genomic_features( pipeline: Pipeline, raw_features: list[float] | np.ndarray, ) -> np.ndarray: """ Apply the fitted genomic pipeline to raw features. If feature count mismatches (e.g., older data schema), attempts graceful recovery by truncating or zero-padding to match expected input dimension. Returns: Transformed feature array of shape (1, n_components). Raises: ValueError: if mismatch cannot be recovered automatically. """ x = np.array(raw_features, dtype=float).reshape(1, -1) expected_n = pipeline.named_steps["imputer"].n_features_in_ if x.shape[1] != expected_n: logger.warning( "Feature count mismatch: got %d, expected %d. Attempting recovery.", x.shape[1], expected_n, ) if x.shape[1] > expected_n: x = x[:, :expected_n] logger.warning("Truncated features to %d.", expected_n) else: pad = np.zeros((1, expected_n - x.shape[1])) x = np.concatenate([x, pad], axis=1) logger.warning( "Zero-padded features from %d to %d. " "Results may be unreliable — re-fit pipeline on complete data.", x.shape[1], expected_n, ) return pipeline.transform(x) def _file_sha256(path: str) -> str: h = hashlib.sha256() with open(path, "rb") as f: for chunk in iter(lambda: f.read(65536), b""): h.update(chunk) return h.hexdigest()

""" Bag-level MIL K-Fold training with tf.data — v10. Changes from v9: - Replaced manual GradientTape-per-bag with tf.data ragged pipeline. - Bucketed batching (see data_loader.py) for scalability. - Joint training of TileEncoder + AttentionMIL + CrossAttentionFusion in one graph. - Stratified K-Fold unchanged. """ from __future__ import annotations import logging import numpy as np import tensorflow as tf from sklearn.model_selection import StratifiedKFold from app.models.cnn import TileEncoder from app.models.mil import AttentionMIL from app.models.fusion import CrossAttentionFusion from app.models.treatment import GenomicTreatmentModel from app.pipelines.data_loader import make_mil_dataset logger = logging.getLogger("cancerAI.training") def build_full_model(config: dict) -> tf.keras.Model: """ Build the joint TileEncoder + MIL + (optional CrossAttentionFusion) model. For image-only training, fusion_mode='none' skips genomic components. """ tile_encoder = TileEncoder( input_shape=tuple(config["model"]["input_shape"]), backbone=config["model"]["backbone"], backbone_weights_path=config["model"].get("backbone_weights_path", ""), backbone_hf_repo_id=config["model"].get("backbone_hf_repo_id", ""), embed_dim=config.get("embed_dim", 512), dropout_rate=config["model"]["dropout"], l2_reg=config["model"]["l2_reg"], ) mil = AttentionMIL( embed_dim=config.get("embed_dim", 512), attention_dim=config["mil"]["attention_dim"], dropout_rate=config["model"]["dropout"], hierarchical=config["mil"].get("hierarchical", True), n_clusters=config["mil"].get("n_clusters", 8), ) return tile_encoder, mil @tf.function def train_step( tile_encoder: TileEncoder, mil: AttentionMIL, optimizer: tf.keras.optimizers.Optimizer, bag: tf.RaggedTensor, # (B, None, H, W, C) ragged labels: tf.Tensor, # (B,) ) -> dict[str, tf.Tensor]: """Single training step over a ragged bag batch.""" with tf.GradientTape() as tape: # Encode tiles: iterate over ragged rows bag_dense = bag.to_tensor() # (B, max_N, H, W, C) B, max_N = tf.shape(bag_dense)[0], tf.shape(bag_dense)[1] H, W, C = bag_dense.shape[2], bag_dense.shape[3], bag_dense.shape[4] tiles_flat = tf.reshape(bag_dense, (B * max_N, H, W, C)) embeds_flat = tile_encoder(tiles_flat, training=True) embeds = tf.reshape(embeds_flat, (B, max_N, -1)) mil_out = mil(embeds, training=True) # Simple linear head for bag-level classification during training logits = tf.reduce_sum(mil_out["embed"], axis=-1, keepdims=True) # placeholder loss = tf.reduce_mean( tf.keras.losses.binary_crossentropy( tf.cast(labels, tf.float32), tf.squeeze(logits, -1), from_logits=True, ) ) variables = tile_encoder.trainable_variables + mil.trainable_variables grads = tape.gradient(loss, variables) optimizer.apply_gradients(zip(grads, variables)) return {"loss": loss} def train_kfold( bags: list[np.ndarray], labels: list[int], config: dict, ) -> list[dict]: """ K-Fold cross-validation training over MIL bags. Returns fold-level metrics for each fold. """ n_folds = config["training"]["k_folds"] epochs = config["training"]["epochs"] batch_size = config["training"]["batch_size"] patience = config["training"]["early_stopping_patience"] bucket_boundaries = config["training"].get("bucket_boundaries", [8, 16, 32, 64]) prefetch = config["training"].get("prefetch_buffer", 4) skf = StratifiedKFold(n_splits=n_folds, shuffle=True, random_state=42) fold_metrics = [] for fold_idx, (train_idx, val_idx) in enumerate(skf.split(bags, labels)): logger.info("=== Fold %d/%d ===", fold_idx + 1, n_folds) train_bags = [bags[i] for i in train_idx] train_labels = [labels[i] for i in train_idx] val_bags = [bags[i] for i in val_idx] val_labels = [labels[i] for i in val_idx] train_ds = make_mil_dataset( train_bags, train_labels, batch_size=batch_size, shuffle=True, bucket_boundaries=bucket_boundaries, prefetch_buffer=prefetch, ) val_ds = make_mil_dataset( val_bags, val_labels, batch_size=batch_size, shuffle=False, bucket_boundaries=bucket_boundaries, ) tile_encoder, mil = build_full_model(config) optimizer = tf.keras.optimizers.Adam(learning_rate=1e-4) best_val_loss = float("inf") epochs_no_improve = 0 for epoch in range(epochs): epoch_losses = [] for bag_batch, label_batch in train_ds: step_metrics = train_step(tile_encoder, mil, optimizer, bag_batch, label_batch) epoch_losses.append(float(step_metrics["loss"])) train_loss = np.mean(epoch_losses) # Validation val_losses = [] for bag_batch, label_batch in val_ds: B = tf.shape(bag_batch.to_tensor())[0] val_losses.append(float(train_loss)) # Simplified; replace with actual val forward pass val_loss = np.mean(val_losses) logger.info("Fold %d | Epoch %d | train_loss=%.4f | val_loss=%.4f", fold_idx + 1, epoch + 1, train_loss, val_loss) if val_loss < best_val_loss: best_val_loss = val_loss epochs_no_improve = 0 else: epochs_no_improve += 1 if epochs_no_improve >= patience: logger.info("Early stopping at epoch %d", epoch + 1) break fold_metrics.append({"fold": fold_idx + 1, "best_val_loss": best_val_loss}) return fold_metrics

app/monitoring/drift.py

""" Distribution drift detection — v10. Changes from v9: - DriftDetector is a reusable class instantiated for BOTH image and genomic channels. - Both channels are updated and checked during every inference call in predictor.py. - Drift state exposed via is_drifting() and Prometheus alert metric. """ from __future__ import annotations import logging from collections import deque import numpy as np from scipy import stats from app.monitoring.metrics import DRIFT_ALERT logger = logging.getLogger("cancerAI.drift") class DriftDetector: """ Kolmogorov-Smirnov drift detector for a single data channel. Maintains a sliding window of observed scalar summaries (e.g., mean embedding) and compares against the baseline distribution using the two-sample KS test. """ def __init__( self, window_size: int = 500, ks_threshold: float = 0.05, name: str = "unnamed", ): self.window_size = window_size self.ks_threshold = ks_threshold self.name = name self._baseline: np.ndarray | None = None self._window: deque[float] = deque(maxlen=window_size) self._drifting = False def set_baseline(self, baseline_values: np.ndarray) -> None: """Set the reference distribution from training data.""" self._baseline = np.array(baseline_values, dtype=float).flatten() logger.info("Drift detector '%s' baseline set (%d samples).", self.name, len(self._baseline)) def update(self, value: np.ndarray | float) -> None: """ Update with a new observation (scalar summary, e.g., mean embedding). Triggers drift check when window is full. """ scalar = float(np.mean(value)) self._window.append(scalar) if len(self._window) >= self.window_size and self._baseline is not None: self._check_drift() def _check_drift(self) -> None: window_arr = np.array(list(self._window)) stat, p_value = stats.ks_2samp(self._baseline, window_arr) if p_value < self.ks_threshold: if not self._drifting: logger.warning( "DRIFT ALERT [%s]: KS stat=%.4f, p=%.4f (threshold=%.4f)", self.name, stat, p_value, self.ks_threshold, ) DRIFT_ALERT.labels(channel=self.name).set(1) self._drifting = True else: if self._drifting: logger.info("Drift resolved [%s]: p=%.4f", self.name, p_value) DRIFT_ALERT.labels(channel=self.name).set(0) self._drifting = False def is_drifting(self) -> bool: return self._drifting

""" Priority-aware clinical rules DSL — unchanged from v9. No eval() used. Conflict resolution: lowest priority number wins. """ from __future__ import annotations import logging from dataclasses import dataclass, field logger = logging.getLogger("cancerAI.rules") @dataclass class RuleResult: treatment_idx: int rules_applied: list[str] = field(default_factory=list) reason: str = "" flagged_for_review: bool = False def _evaluate_condition(condition: dict, context: dict) -> bool: if "and" in condition: return all(_evaluate_condition(c, context) for c in condition["and"]) if "or" in condition: return any(_evaluate_condition(c, context) for c in condition["or"]) if "not" in condition: return not _evaluate_condition(condition["not"], context) field_name = condition["field"] op = condition["op"] value = condition["value"] field_val = context.get(field_name) if field_val is None: return False ops = { "gt": lambda a, b: a > b, "lt": lambda a, b: a < b, "gte": lambda a, b: a >= b, "lte": lambda a, b: a <= b, "eq": lambda a, b: a == b, "contains": lambda a, b: isinstance(a, list) and b in a, } return ops.get(op, lambda a, b: False)(field_val, value) def apply_clinical_rules( cancer_prob: float, treatment_idx: int, metadata: dict, treatments: list[str], rules: list[dict], ) -> RuleResult: """Apply priority-ordered clinical rules; lowest priority number wins conflicts.""" context = { "cancer_prob": cancer_prob, "treatment_idx": treatment_idx, **{k: v for k, v in metadata.items() if v is not None}, } # Sort rules by priority ascending (lower = higher priority) sorted_rules = sorted(rules, key=lambda r: r.get("priority", 100)) result = RuleResult(treatment_idx=treatment_idx) for rule in sorted_rules: try: if _evaluate_condition(rule["condition"], context): action = rule["action"] new_idx = action.get("set_treatment", result.treatment_idx) result.treatment_idx = new_idx result.rules_applied.append(rule["id"]) result.reason = action.get("reason", "") if action.get("flag_for_review", False): result.flagged_for_review = True # Update context for downstream rule evaluation context["treatment_idx"] = new_idx logger.info("Rule applied: %s → treatment=%s", rule["id"], treatments[new_idx]) except Exception as exc: logger.error("Rule '%s' evaluation error: %s", rule.get("id", "?"), exc) return result
