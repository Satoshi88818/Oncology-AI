""" Request-ID correlation middleware — v12 NEW. Attaches a unique X-Request-ID header to every request and response so that audit log entries, Prometheus metrics, and application logs can be correlated across services by a single trace identifier. """ from __future__ import annotations import uuid import logging from contextvars import ContextVar from starlette.middleware.base import BaseHTTPMiddleware from starlette.requests import Request from starlette.responses import Response logger = logging.getLogger("cancerAI.middleware") # Context variable makes the request_id accessible anywhere in the call stack # without threading — safe with async and sync handlers. request_id_ctx: ContextVar[str] = ContextVar("request_id", default="") def get_request_id() -> str: return request_id_ctx.get("") class RequestIDMiddleware(BaseHTTPMiddleware): """ Reads X-Request-ID from the incoming request (set by upstream proxy/gateway) or generates a fresh UUID4 if absent. Stores it in the context variable and echoes it back in the response header. """ async def dispatch(self, request: Request, call_next) -> Response: req_id = request.headers.get("X-Request-ID") or str(uuid.uuid4()) token = request_id_ctx.set(req_id) try: response: Response = await call_next(request) response.headers["X-Request-ID"] = req_id return response finally: request_id_ctx.reset(token)

""" ONNX Runtime model runner — v12 NEW. Wraps onnxruntime.InferenceSession for each model component. Used by predictor.py when config.inference.use_onnx = true. """ from __future__ import annotations import logging from pathlib import Path from typing import Any import numpy as np logger = logging.getLogger("cancerAI.onnx_runner") def _get_providers() -> list[str]: """Return ONNX Runtime execution providers in priority order.""" try: import onnxruntime as ort available = ort.get_available_providers() # Prefer TensorRT > CUDA > CPU preferred = ["TensorrtExecutionProvider", "CUDAExecutionProvider", "CPUExecutionProvider"] return [p for p in preferred if p in available] or ["CPUExecutionProvider"] except ImportError: return ["CPUExecutionProvider"] class OnnxModelRunner: """ Lightweight wrapper around an ONNX Runtime inference session. Usage: runner = OnnxModelRunner("models/tile_encoder.onnx") outputs = runner.run({"input": tile_array}) embed = outputs["output"] """ def __init__(self, model_path: str): import onnxruntime as ort path = Path(model_path) if not path.exists(): raise FileNotFoundError( f"ONNX model not found at '{model_path}'. " "Run scripts/export_onnx.py to generate ONNX models." ) providers = _get_providers() logger.info("Loading ONNX model: %s (providers=%s)", path.name, providers) self._session = ort.InferenceSession(str(path), providers=providers) self._input_names = [i.name for i in self._session.get_inputs()] self._output_names = [o.name for o in self._session.get_outputs()] logger.info( "ONNX session ready: inputs=%s outputs=%s", self._input_names, self._output_names, ) def run(self, inputs: dict[str, np.ndarray]) -> dict[str, np.ndarray]: """ Run inference. Args: inputs: dict mapping input name → numpy array. Returns: dict mapping output name → numpy array. """ feed = {k: v.astype(np.float32) if v.dtype != np.float32 else v for k, v in inputs.items()} raw_outputs = self._session.run(self._output_names, feed) return dict(zip(self._output_names, raw_outputs)) def run_mc_dropout( self, inputs: dict[str, np.ndarray], n_samples: int, output_key: str, ) -> np.ndarray: """ Approximate MC Dropout by tiling the batch dimension and running a single ONNX session call (requires dropout nodes exported with training=True). Returns: array of shape (n_samples, batch_size, ...) """ # Tile input along batch axis tiled_inputs = { k: np.concatenate([v] * n_samples, axis=0) for k, v in inputs.items() } tiled_outputs = self.run(tiled_inputs) out = tiled_outputs[output_key] # (n_samples * B, ...) B = list(inputs.values())[0].shape[0] return out.reshape(n_samples, B, *out.shape[1:]) 

app/pipelines/wsi.py

""" Adaptive WSI tiling with quality filtering — v12. NEW: Tile quality filtering pipeline: - Laplacian variance check (blur detection) - Brightness/contrast statistics (artifact/out-of-focus rejection) - Staining artifact detection via saturation standard deviation Returns tiles as numpy arrays with quality metadata. """ from __future__ import annotations import logging from concurrent.futures import ThreadPoolExecutor from typing import NamedTuple import cv2 import numpy as np from PIL import Image logger = logging.getLogger("cancerAI.wsi") class TileWithMeta(NamedTuple): tile: np.ndarray # (H, W, C) float32, normalised to [0, 1] x: int # tile x coordinate in slide pixels (level 0) y: int # tile y coordinate in slide pixels (level 0) cellularity: float laplacian_variance: float mean_brightness: float def _compute_tissue_ratio(tile_np: np.ndarray) -> float: """Estimate tissue fraction via HSV saturation thresholding.""" hsv = cv2.cvtColor((tile_np * 255).astype(np.uint8), cv2.COLOR_RGB2HSV) saturation = hsv[:, :, 1] tissue_mask = saturation > 20 return float(tissue_mask.mean()) def _compute_laplacian_variance(tile_np: np.ndarray) -> float: """Measure tile sharpness via Laplacian variance. Low = blurry.""" gray = cv2.cvtColor((tile_np * 255).astype(np.uint8), cv2.COLOR_RGB2GRAY) return float(cv2.Laplacian(gray, cv2.CV_64F).var()) def _compute_brightness_stats(tile_np: np.ndarray) -> tuple[float, float]: """Return (mean_brightness, saturation_std) in [0, 255] scale.""" img_uint8 = (tile_np * 255).astype(np.uint8) hsv = cv2.cvtColor(img_uint8, cv2.COLOR_RGB2HSV) brightness = float(hsv[:, :, 2].mean()) saturation_std = float(hsv[:, :, 1].std()) return brightness, saturation_std def _passes_quality_filter( tile_np: np.ndarray, min_laplacian_variance: float, min_brightness: float, max_brightness: float, max_saturation_std: float, ) -> tuple[bool, dict]: """ Return (passes, stats_dict). Rejection criteria: - Blurry tile: Laplacian variance < min_laplacian_variance - Too dark (out-of-focus): mean brightness < min_brightness - Too bright (background glass/empty): mean brightness > max_brightness - Low saturation variance (staining artifact or background): sat_std > max_saturation_std """ lap_var = _compute_laplacian_variance(tile_np) brightness, sat_std = _compute_brightness_stats(tile_np) stats = { "laplacian_variance": lap_var, "mean_brightness": brightness, "saturation_std": sat_std, } if lap_var < min_laplacian_variance: return False, stats if brightness < min_brightness or brightness > max_brightness: return False, stats if sat_std > max_saturation_std: return False, stats return True, stats def extract_tiles( slide_path: str, tile_size: int = 512, level: int = 0, min_tissue_ratio: float = 0.08, max_tiles: int = 100, max_workers: int = 8, adaptive_sampling: bool = True, high_cellularity_threshold: float = 0.6, adaptive_max_tiles: int = 128, quality_filter_config: dict | None = None, return_metadata: bool = False, ) -> list[np.ndarray] | list[TileWithMeta]: """ Extract tissue tiles from a WSI with quality filtering. Args: quality_filter_config: dict with keys matching QualityFilterConfig fields. Pass None to disable quality filtering. return_metadata: if True, returns list[TileWithMeta] with coordinates. Returns: List of (H, W, C) float32 numpy arrays (normalised to [0, 1]), or list of TileWithMeta if return_metadata=True. """ try: import openslide # type: ignore except ImportError: raise RuntimeError("openslide-python not installed.") qf = quality_filter_config or {} filter_enabled = qf.get("enabled", False) min_lap = qf.get("min_laplacian_variance", 100.0) min_bright = qf.get("min_brightness", 40.0) max_bright = qf.get("max_brightness", 230.0) max_sat_std = qf.get("max_saturation_std", 80.0) slide = openslide.OpenSlide(slide_path) width, height = slide.level_dimensions[level] step = tile_size candidate_coords = [ (x, y) for y in range(0, height - tile_size + 1, step) for x in range(0, width - tile_size + 1, step) ] tiles_meta: list[TileWithMeta] = [] rejected_count = 0 def process_coord(coord): x, y = coord region = slide.read_region((x, y), level, (tile_size, tile_size)) tile_np = np.array(region.convert("RGB"), dtype=np.float32) / 255.0 tissue_ratio = _compute_tissue_ratio(tile_np) if tissue_ratio < min_tissue_ratio: return None if filter_enabled: passes, stats = _passes_quality_filter( tile_np, min_lap, min_bright, max_bright, max_sat_std ) if not passes: return "rejected" lap_var = stats["laplacian_variance"] brightness = stats["mean_brightness"] else: lap_var = 0.0 brightness = 0.0 return TileWithMeta( tile=tile_np, x=x, y=y, cellularity=tissue_ratio, laplacian_variance=lap_var, mean_brightness=brightness, ) with ThreadPoolExecutor(max_workers=max_workers) as executor: results = list(executor.map(process_coord, candidate_coords)) for r in results: if r is None: continue if r == "rejected": rejected_count += 1 continue tiles_meta.append(r) if rejected_count > 0: logger.info("Quality filter rejected %d tiles from %s", rejected_count, slide_path) if not tiles_meta: logger.warning("No usable tiles extracted from %s", slide_path) return [] # Adaptive sampling: oversample high-cellularity regions if adaptive_sampling: high_cell = [t for t in tiles_meta if t.cellularity >= high_cellularity_threshold] low_cell = [t for t in tiles_meta if t.cellularity < high_cellularity_threshold] n_high = min(len(high_cell), adaptive_max_tiles * 2 // 3) n_low = min(len(low_cell), adaptive_max_tiles - n_high) tiles_meta = high_cell[:n_high] + low_cell[:n_low] else: tiles_meta = tiles_meta[:max_tiles] if return_metadata: return tiles_meta return [t.tile for t in tiles_meta]

""" Multivariate distribution drift detection — v12. REPLACED: scalar KS test → MMD (Maximum Mean Discrepancy) on full embedding vectors + per-dimension KS alerting. NEW: Data quality monitoring (blur stats, staining artifact tracking). """ from __future__ import annotations import logging from collections import deque from pathlib import Path from typing import Optional import numpy as np from scipy import stats as scipy_stats logger = logging.getLogger("cancerAI.drift") def _rbf_kernel(X: np.ndarray, Y: np.ndarray, sigma: float = 1.0) -> float: """ Compute unbiased MMD² estimate using RBF kernel. MMD² = E[k(x,x')] - 2·E[k(x,y)] + E[k(y,y')] where k(a,b) = exp(-||a-b||² / (2σ²)) """ def kernel(A: np.ndarray, B: np.ndarray) -> float: dists = np.sum((A[:, None] - B[None, :]) ** 2, axis=-1) return float(np.exp(-dists / (2 * sigma ** 2)).mean()) return kernel(X, X) - 2 * kernel(X, Y) + kernel(Y, Y) class DriftDetector: """ Multivariate drift detector using MMD on full embedding vectors. Falls back to scalar KS for backward compatibility when embeddings are 1-D. Raises alerts when: 1. MMD² between baseline and current window exceeds `mmd_threshold` 2. Any per-dimension KS test p-value < `ks_threshold` 3. Any per-dimension mean shift exceeds `per_dim_threshold` """ def __init__( self, window_size: int = 500, ks_threshold: float = 0.05, mmd_threshold: float = 0.01, per_dim_threshold: float = 0.1, name: str = "unnamed", ): self._window_size = window_size self._ks_threshold = ks_threshold self._mmd_threshold = mmd_threshold self._per_dim_threshold = per_dim_threshold self._name = name self._window: deque[np.ndarray] = deque(maxlen=window_size) self._baseline: Optional[np.ndarray] = None # (N_baseline, D) self._is_drifting = False def set_baseline(self, samples: np.ndarray) -> None: """Set baseline from an array of shape (N, D) or (N,) for scalar case.""" if samples.ndim == 1: samples = samples[:, None] self._baseline = samples logger.info( "DriftDetector[%s]: baseline set from %d samples, dim=%d", self._name, len(samples), samples.shape[1], ) def set_baseline_from_file(self, path: str) -> None: if not path or not Path(path).exists(): logger.warning( "DriftDetector[%s]: baseline file not found at '%s'. " "Drift detection disabled until baseline is set.", self._name, path, ) return arr = np.load(path) self.set_baseline(arr) def update(self, embedding: np.ndarray) -> None: """ Add a new embedding vector (shape (D,) or scalar) to the sliding window and check for drift. """ if embedding.ndim == 0: embedding = np.array([float(embedding)]) elif embedding.ndim > 1: embedding = embedding.flatten() self._window.append(embedding) if len(self._window) >= self._window_size // 2: self._check_drift() def _check_drift(self) -> None: if self._baseline is None: return window_arr = np.stack(list(self._window)) # (W, D) baseline = self._baseline # (N, D) if window_arr.ndim == 1: window_arr = window_arr[:, None] if baseline.ndim == 1: baseline = baseline[:, None] # Sub-sample baseline to match window size for efficiency n_baseline = min(len(baseline), len(window_arr) * 2) idx = np.random.choice(len(baseline), size=n_baseline, replace=False) baseline_sample = baseline[idx] # 1. MMD test (multivariate) mmd_score = _rbf_kernel(window_arr, baseline_sample) if mmd_score > self._mmd_threshold: logger.warning( "DRIFT ALERT [%s]: MMD²=%.4f > threshold=%.4f", self._name, mmd_score, self._mmd_threshold, ) self._is_drifting = True return # 2. Per-dimension KS test n_dims = window_arr.shape[1] drift_dims = [] for d in range(n_dims): ks_stat, p_val = scipy_stats.ks_2samp(baseline_sample[:, d], window_arr[:, d]) mean_shift = abs(window_arr[:, d].mean() - baseline_sample[:, d].mean()) if p_val < self._ks_threshold or mean_shift > self._per_dim_threshold: drift_dims.append(d) if drift_dims: logger.warning( "DRIFT ALERT [%s]: per-dimension drift in %d/%d dimensions: %s", self._name, len(drift_dims), n_dims, drift_dims[:10], ) self._is_drifting = True else: self._is_drifting = False def is_drifting(self) -> bool: return self._is_drifting @property def window_size(self) -> int: return len(self._window)

tf.reduce_mean( tf.keras.losses.binary_crossentropy( tf.cast(labels, tf.float32), tf.squeeze(head_out["logits"], -1), from_logits=True, ) ) return float(loss.numpy()), head_out["probs"].numpy().squeeze(), labels.numpy() def train_kfold( bags: list, labels: list[int], config: dict, save_dir: str = "models", ) -> list[dict]: """ K-Fold cross-validation training. NEW: Mixed precision via config.training.mixed_precision. NEW: Distributed training via config.training.distributed (MirroredStrategy). """ from sklearn.metrics import roc_auc_score use_mixed_precision = config.get("training", {}).get("mixed_precision", False) use_distributed = config.get("training", {}).get("distributed", False) if config["training"].get("bucket_boundaries", [8, 16, 32, 64]) prefetch = config["training"].get("prefetch_buffer", 4) skf = StratifiedKFold(n_splits=n_folds, shuffle=True, random_state=42) fold_metrics [labels[i] for i in train_idx] val_bags = [bags[i] for i in val_idx] val_labels = [labels[i] for i in val_idx] train_ds = make_mil_dataset( train_bags, train_labels, batch_size=batch_size, shuffle=True, bucket_boundaries=bucket_boundaries, prefetch_buffer=prefetch, ) val_ds = make_mil_dataset( val_bags, val_labels, batch_size=batch_size, shuffle=False, bucket_boundaries=bucket_boundaries, ) with strategy.scope(): tile_encoder, mil, cancer_head = build_full_model(config) optimizer = tf.keras.optimizers.Adam(learning_rate=1e-4) best_val_loss = float("inf") epochs_no_improve = 0 for epoch in range(epochs): train_losses = [] for bag_batch, label_batch in train_ds: loss = _train_step(tile_encoder, mil, cancer_head, optimizer, bag_batch, label_batch) train_losses.append(float(loss.numpy())) train_loss = np.mean(train_losses) val_losses, all_probs, all_labels = [], [], [] for bag_batch, label_batch in val_ds: v_loss, probs, lbls = _val_step(tile_encoder, mil, cancer_head, bag_batch, label_batch) val_losses.append(v_loss) all_probs.extend(probs.flatten().tolist()) all_labels.extend(lbls.flatten().tolist()) val_loss = fold_idx + 1, epoch + 1, train_loss, val_loss, val_auc, ) if val_loss < best_val_loss: best_val_loss = val_loss epochs_no_improve = 0 tile_encoder.save_weights(f"{save_dir}/fold{fold_idx}_tile_encoder.weights.h5") mil.save_weights(f"{save_dir}/fold{fold_idx}_mil.weights.h5") cancer_head.save_weights(f"{save_dir}/fold{fold_idx}_cancer_head.weights.h5") else: epochs_no_improve += 1 if epochs_no_improve >= patience: logger.info("Early stopping at epoch %d", epoch + 1) break fold_metrics.append({ "fold": fold_idx + 1, "best_val_loss": best_val_loss, "final_val_auc": val_auc, }) if use_mixed_precision: tf.keras.mixed_precision.set_global_policy("float32") return """ scripts/validate_onnx.py — NEW v12. Validates ONNX model outputs against Keras outputs for numerical parity. Should be run as part of CI/CD before any production deployment.
