" "CancerAI will not start without an encryption key. " "Generate one with: python -c \"from cryptography.fernet import Fernet; print(Fernet.generate_key().decode())\"" ) return key.encode() if isinstance(key, str) else key _cipher = Fernet(_load_encryption_key()) def encrypt(plaintext: str) -> str: """Encrypt a plaintext string, returning a URL-safe base64 token.""" return _cipher.encrypt(plaintext.encode()).decode() def decrypt(token: str) -> str: """Decrypt a Fernet token back to plaintext.""" return _cipher.decrypt(token.encode()).decode() 

app/rules/engine.py

""" Safe clinical rules engine — no eval(), no exec(). Rules are expressed as structured condition trees (parsed from YAML config). Operators: gt, lt, gte, lte, eq, contains, and, or, not Fields: cancer_prob (float), treatment_idx (int), age (int), comorbidities (list[str]) """ from __future__ import annotations import logging from dataclasses import dataclass, field from typing import Any logger = logging.getLogger("cancerAI.rules") @dataclass class RuleResult: treatment_idx: int description: str reason: str flagged_for_review: bool = False rules_applied: list[str] = field(default_factory=list) def _evaluate_condition(condition: dict, context: dict) -> bool: """ Recursively evaluate a condition tree without eval(). Supported structure: {"and": [cond1, cond2, ...]} {"or": [cond1, cond2, ...]} {"not": cond} {"field": "cancer_prob", "op": "gt", "value": 0.85} """ if "and" in condition: return all(_evaluate_condition(c, context) for c in condition["and"]) if "or" in condition: return any(_evaluate_condition(c, context) for c in condition["or"]) if "not" in condition: return not _evaluate_condition(condition["not"], context) field_name = condition.get("field") op = condition.get("op") value = condition.get("value") if field_name not in context: logger.warning("Rules engine: unknown field '%s' — condition skipped", field_name) return False actual = context[field_name] try: match op: case "gt": return float(actual) > float(value) case "lt": return float(actual) < float(value) case "gte": return float(actual) >= float(value) case "lte": return float(actual) <= float(value) case "eq": return actual == value case "contains": if not isinstance(actual, (list, str)): return False return value in actual case _: logger.warning("Rules engine: unknown operator '%s' — condition skipped", op) return False except (TypeError, ValueError) as e: logger.warning("Rules engine: comparison error for field '%s': %s", field_name, e) return False def apply_clinical_rules( cancer_prob: float, treatment_idx: int, metadata: dict, treatments: list[str], rules: list[dict], ) -> RuleResult: """ Apply ordered clinical rules to post-process model output. Rules are evaluated in declaration order. First matching rule wins. No eval() is used at any point. """ context = { "cancer_prob": cancer_prob, "treatment_idx": treatment_idx, "age": metadata.get("age", 0), "comorbidities": metadata.get("comorbidities", []), } applied: list[str] = [] flagged = False for rule in rules: rule_id = rule.get("id", "unnamed") try: if _evaluate_condition(rule["condition"], context): action = rule["action"] new_idx = action.get("set_treatment", treatment_idx) reason = action.get("reason", "Clinical rule applied") if action.get("flag_for_review"): flagged = True applied.append(rule_id) logger.info("Rule '%s' matched → treatment %d → %d", rule_id, treatment_idx, new_idx) # Update context for subsequent rules context["treatment_idx"] = new_idx treatment_idx = new_idx except Exception as e: logger.error("Rule '%s' evaluation error: %s — skipped", rule_id, e) continue desc = treatments[treatment_idx] if 0 <= treatment_idx < len(treatments) else "Unknown" return RuleResult( treatment_idx=treatment_idx, description=desc, reason=reason if applied else "Model recommendation (no rules matched)", flagged_for_review=flagged, rules_applied=applied, ) 

app/pipelines/wsi.py

""" Whole Slide Image tiling pipeline — production-grade tissue detection. Improvements over v7: - Returns tile coordinates alongside arrays (for MIL spatial context) - Hard cap on max_tiles to prevent OOM - Explicit resource cleanup with context manager - Configurable HSV thresholds """ from __future__ import annotations import logging from dataclasses import dataclass import cv2 import numpy as np import openslide logger = logging.getLogger("cancerAI.wsi") @dataclass class Tile: array: np.ndarray # float32, shape (H, W, 3), normalised 0–1 x: int y: int tissue_ratio: float def extract_tiles( slide_path: str, tile_size: int = 512, level: int = 0, min_tissue_ratio: float = 0.08, max_tiles: int = 200, target_shape: tuple[int, int] = (300, 300), saturation_threshold: int = 30, value_threshold: int = 50, ) -> list[Tile]: """ Extract tissue-containing tiles from a WSI. Args: slide_path: Path to .svs / .tiff / .ndpi slide. tile_size: Tile edge length in pixels at the specified level. level: OpenSlide pyramid level (0 = highest resolution). min_tissue_ratio: Minimum fraction of pixels classified as tissue. max_tiles: Hard cap — prevents OOM on very large slides. target_shape: Resize each tile to (H, W) for model input. saturation_threshold: HSV saturation threshold for tissue detection. value_threshold: HSV value (brightness) threshold for tissue detection. Returns: List of Tile objects sorted by tissue_ratio descending (best tiles first). """ tiles: list[Tile] = [] with openslide.OpenSlide(slide_path) as slide: width, height = slide.level_dimensions[level] logger.info("WSI dimensions at level %d: %d × %d px", level, width, height) for y in range(0, height, tile_size): for x in range(0, width, tile_size): if len(tiles) >= max_tiles: logger.info("Reached max_tiles cap (%d) — stopping early", max_tiles) break tile_img = slide.read_region((x, y), level, (tile_size, tile_size)) tile_rgb = np.array(tile_img.convert("RGB"), dtype=np.uint8) hsv = cv2.cvtColor(tile_rgb, cv2.COLOR_RGB2HSV) tissue_mask = (hsv[:, :, 1] > saturation_threshold) & (hsv[:, :, 2] > value_threshold) tissue_ratio = float(np.mean(tissue_mask)) if tissue_ratio >= min_tissue_ratio: resized = cv2.resize(tile_rgb, (target_shape[1], target_shape[0])) tiles.append(Tile( array=(resized.astype(np.float32) / 255.0), x=x, y=y, tissue_ratio=tissue_ratio, )) if len(tiles) >= max_tiles: break tiles.sort(key=lambda t: t.tissue_ratio, reverse=True) logger.info("Extracted %d high-quality tiles from WSI", len(tiles)) return tiles 

app/models/mil.py

""" Attention-based Multiple Instance Learning (MIL) aggregator. Replaces the naive tile mean from v7 with a learnable attention mechanism. Based on: Ilse et al. (2018) "Attention-based Deep Multiple Instance Learning" https://arxiv.org/abs/1802.04712 Each tile produces a feature vector. The MIL head learns which tiles are most diagnostically relevant and weights them accordingly — preserving spatial context that a simple mean destroys. """ from __future__ import annotations import tensorflow as tf from tensorflow.keras import layers, models def build_mil_aggregator( feature_dim: int = 1280, # EfficientNetB3 GlobalAvgPool output attention_dim: int = 128, n_classes: int = 1, dropout: float = 0.3, ) -> tf.keras.Model: """ Build an attention MIL aggregator. Input: (batch, n_tiles, feature_dim) — variable n_tiles handled via masking Output: (batch, n_classes) — bag-level prediction """ tile_features = layers.Input(shape=(None, feature_dim), name="tile_features") # Attention scoring: Tanh gate × Sigmoid gate (Gated Attention) # Shape: (batch, n_tiles, attention_dim) V = layers.TimeDistributed(layers.Dense(attention_dim, activation="tanh"), name="attention_V")(tile_features) U = layers.TimeDistributed(layers.Dense(attention_dim, activation="sigmoid"), name="attention_U")(tile_features) gated = layers.Multiply(name="gated_attention")([V, U]) # Attention weights: (batch, n_tiles, 1) → softmax over tiles raw_scores = layers.TimeDistributed(layers.Dense(1), name="attention_scores")(gated) attention_weights = layers.Softmax(axis=1, name="attention_softmax")(raw_scores) # Weighted aggregation: (batch, feature_dim) bag_embedding = layers.Dot(axes=1, name="weighted_pool")([ layers.Permute((2, 1))(attention_weights), tile_features ]) bag_embedding = layers.Flatten()(bag_embedding) # Classification head x = layers.Dense(256, activation="relu")(bag_embedding) x = layers.Dropout(dropout)(x, training=True) # Keep training=True for MC dropout output = layers.Dense(n_classes, activation="sigmoid", name="cancer_probability")(x) return models.Model(inputs=tile_features, outputs=output, name="MIL_Aggregator")

""" EfficientNetB3 feature extractor for histopathology tiles. FIX from v7: base.trainable and the training= argument are now consistent. - During feature extraction (inference): base.trainable = False, training=False → BatchNorm uses running statistics (correct behaviour) - During fine-tuning (optional): base.trainable = True, training=True PRODUCTION NOTE: Consider replacing EfficientNetB3 (ImageNet pretrained) with a histopathology-specific backbone: - UNI (https://huggingface.co/MahmoodLab/uni) - CONCH (https://github.com/mahmoodlab/CONCH) - PLIP (https://github.com/PathologyFoundation/plip) These domain-adapted models consistently outperform ImageNet backbones on H&E image tasks. """ from __future__ import annotations import tensorflow as tf from tensorflow.keras import applications, layers, models def build_tile_feature_extractor( input_shape: tuple[int, int, int] = (300, 300, 3), fine_tune: bool = False, ) -> tf.keras.Model: """ EfficientNetB3 tile feature extractor. Returns a model that maps (batch, H, W, 3) → (batch, feature_dim). The output feeds the MIL aggregator. Args: input_shape: (H, W, C) — must match WSI tile target_shape. fine_tune: If True, backbone weights are unfrozen for end-to-end training. """ base = applications.EfficientNetB3( include_top=False, weights="imagenet", input_shape=input_shape, ) base.trainable = fine_tune # Consistent: trainable flag matches training= kwarg inputs = layers.Input(shape=input_shape, name="tile_input") # training=fine_tune ensures BN uses running stats during frozen inference x = base(inputs, training=fine_tune) features = layers.GlobalAveragePooling2D(name="gap")(x) return models.Model(inputs, features, name="TileFeatureExtractor") 

app/models/treatment.py

""" Genomic MLP for treatment recommendation. Improvements over v7: - Stratified K-Fold used at training time (see pipelines/training.py) - MC Dropout enabled at inference time (Dropout layers with training=True) - Output includes per-class probabilities for uncertainty estimation """ from __future__ import annotations import tensorflow as tf from tensorflow.keras import layers, models, regularizers def build_treatment_model( n_genomic_features: int = 128, n_treatments: int = 6, dropout: float = 0.3, l2_reg: float = 1e-4, ) -> tf.keras.Model: """ 3-layer MLP for treatment recommendation from genomic features. MC Dropout: Dropout layers always run with training=True — this means inference calls with different random seeds produce a distribution of outputs, enabling uncertainty quantification via repeated sampling. """ inputs = layers.Input(shape=(n_genomic_features,), name="genomic_features") x = layers.Dense(256, kernel_regularizer=regularizers.l2(l2_reg), name="dense_1")(inputs) x = layers.BatchNormalization(name="bn_1")(x) x = layers.Activation("relu")(x) x = layers.Dropout(dropout, name="drop_1")(x, training=True) x = layers.Dense(128, kernel_regularizer=regularizers.l2(l2_reg), name="dense_2")(x) x = layers.BatchNormalization(name="bn_2")(x) x = layers.Activation("relu")(x) x = layers.Dropout(dropout, name="drop_2")(x, training=True) outputs = layers.Dense(n_treatments, activation="softmax", name="treatment_probs")(x) model = models.Model(inputs, outputs, name="TreatmentMLP") model.compile( optimizer="adam", loss="sparse_categorical_crossentropy", metrics=["accuracy"], ) return model

""" Training pipeline for CancerAI v8. Key improvements over v7: 1. StratifiedKFold instead of KFold — preserves class balance across folds (critical for imbalanced oncology datasets). 2. Dataset is NOT materialised into RAM if it exceeds a safe threshold — uses TFRecord-based approach for large cohorts. 3. Separate StandardScaler fitted per fold on training split only (v7 fitted globally, risking subtle leakage). 4. Class weights computed per fold for imbalanced label handling. 5. Full training loop returned — no truncated callbacks. """ from __future__ import annotations import logging import os import joblib import numpy as np import tensorflow as tf from sklearn.model_selection import StratifiedKFold from sklearn.preprocessing import StandardScaler from sklearn.utils.class_weight import compute_class_weight from tensorflow.keras import callbacks from app.models.cnn import build_tile_feature_extractor from app.models.mil import build_mil_aggregator from app.models.treatment import build_treatment_model logger = logging.getLogger("cancerAI.training") # Hard limit before switching to TFRecord streaming _RAM_MATERIALIZE_LIMIT = 50_000 def train( dataset: tf.data.Dataset, model_dir: str, k_folds: int = 5, batch_size: int = 32, epochs: int = 50, early_stopping_patience: int = 7, input_shape: tuple[int, int, int] = (300, 300, 3), n_genomic_features: int = 128, n_treatments: int = 6, ) -> dict: """ Full K-Fold training pipeline. Returns a dict of per-fold validation metrics. """ os.makedirs(model_dir, exist_ok=True) logger.info("Materialising dataset for K-Fold training...") data = list(dataset.as_numpy_iterator()) if len(data) > _RAM_MATERIALIZE_LIMIT: raise RuntimeError( f"Dataset size ({len(data)} samples) exceeds RAM materialisation limit " f"({_RAM_MATERIALIZE_LIMIT}). Convert your dataset to TFRecord format and " "use a streaming training loop instead." ) images = np.stack([d["image"] for d in data]) labels = np.array([d["label"] for d in data]) genomics = np.stack([d["genomic"] for d in data]) treatments = np.array([d["treatment"] for d in data]) # StratifiedKFold preserves class balance skf = StratifiedKFold(n_splits=k_folds, shuffle=True, random_state=42) fold_metrics: dict[int, dict] = {} for fold, (train_idx, val_idx) in enumerate(skf.split(images, labels)): logger.info("━━━ Fold %d / %d ━━━", fold + 1, k_folds) # Scale genomic features on train split only (no leakage) scaler = StandardScaler() genomics_train = scaler.fit_transform(genomics[train_idx]) genomics_val = scaler.transform(genomics[val_idx]) if fold == k_folds - 1: # Save the final fold's scaler as the production scaler joblib.dump(scaler, os.path.join(model_dir, "scaler.pkl")) logger.info("Saved production scaler from fold %d", fold + 1) # Class weights for imbalanced labels unique_classes = np.unique(labels[train_idx]) cw_array = compute_class_weight("balanced", classes=unique_classes, y=labels[train_idx]) class_weights = dict(zip(unique_classes, cw_array)) # ── CNN + MIL ──────────────────────────────────────────────────────── extractor = build_tile_feature_extractor(input_shape=input_shape, fine_tune=False) feature_dim = extractor.output_shape[-1] mil = build_mil_aggregator(feature_dim=feature_dim) # Extract features (frozen backbone — fast) img_train_feats = extractor.predict(images[train_idx], batch_size=batch_size, verbose=0) img_val_feats = extractor.predict(images[val_idx], batch_size=batch_size, verbose=0) # Expand to (batch, 1, feature_dim) — single "tile" per patch image # (For real WSIs, feed (batch, n_tiles, feature_dim) from wsi.py) img_train_feats = np.expand_dims(img_train_feats, axis=1) img_val_feats = np.expand_dims(img_val_feats, axis=1) mil.compile(optimizer="adam", loss="binary_crossentropy", metrics=["accuracy", tf.keras.metrics.AUC(name="auc")]) cnn_callbacks = [ callbacks.EarlyStopping(monitor="val_auc", patience=early_stopping_patience, restore_best_weights=True, mode="max"), callbacks.ModelCheckpoint( os.path.join(model_dir, f"mil_fold{fold}.keras"), monitor="val_auc", save_best_only=True, mode="max", ), callbacks.ReduceLROnPlateau(monitor="val_auc", factor=0.5, patience=3, mode="max"), ] cnn_hist = mil.fit( img_train_feats, labels[train_idx], validation_data=(img_val_feats, labels[val_idx]), batch_size=batch_size, epochs=epochs, callbacks=cnn_callbacks, class_weight=class_weights, verbose=1, ) # ── Treatment MLP ───────────────────────────────────────────────────── treatment_model = build_treatment_model( n_genomic_features=n_genomic_features, n_treatments=n_treatments, ) # Class weights for treatments unique_trt = np.unique(treatments[train_idx]) trt_cw = compute_class_weight("balanced", classes=unique_trt, y=treatments[train_idx]) trt_class_weights = dict(zip(unique_trt, trt_cw)) trt_callbacks = [ callbacks.EarlyStopping(monitor="val_accuracy", patience=early_stopping_patience, restore_best_weights=True), callbacks.ModelCheckpoint( os.path.join(model_dir, f"treatment_fold{fold}.keras"), monitor="val_accuracy", save_best_only=True, ), ] trt_hist = treatment_model.fit( genomics_train, treatments[train_idx], validation_data=(genomics_val, treatments[val_idx]), batch_size=batch_size, epochs=epochs, callbacks=trt_callbacks, class_weight=trt_class_weights, verbose=1, ) fold_metrics[fold] = { "cnn_val_auc": max(cnn_hist.history.get("val_auc", [0])), "treatment_val_accuracy": max(trt_hist.history.get("val_accuracy", [0])), } logger.info("Fold %d — CNN AUC: %.4f | Treatment Acc: %.4f", fold + 1, fold_metrics[fold]["cnn_val_auc"], fold_metrics[fold]["treatment_val_accuracy"]) # Save best fold models as production models _promote_best_fold(fold_metrics, model_dir, k_folds) return fold_metrics def _promote_best_fold(fold_metrics: dict, model_dir: str, k_folds: int): """Copy best fold models to production paths.""" import shutil best_fold = max(fold_metrics, key=lambda f: fold_metrics[f]["cnn_val_auc"]) logger.info("Best fold: %d — promoting to production models", best_fold + 1) shutil.copy( os.path.join(model_dir, f"mil_fold{best_fold}.keras"), os.path.join(model_dir, "mil_production.keras"), ) shutil.copy( os.path.join(model_dir, f"treatment_fold{best_fold}.keras"), os.path.join(model_dir, "treatment_production.keras"), ) 

app/inference/predictor.py

""" CancerAI v8 inference engine. Features: - Monte-Carlo Dropout uncertainty estimation (N forward passes with dropout active) - SHAP DeepExplainer for feature attribution on genomic predictions - Clinical rules engine post-processing (no eval) - Full clinical disclaimer in every result """ from __future__ import annotations import json import logging import os import joblib import numpy as np import tensorflow as tf from PIL import Image from app.inference.explainer import compute_shap_explanations from app.models.cnn import build_tile_feature_extractor from app.models.mil import build_mil_aggregator from app.models.treatment import build_treatment_model from app.pipelines.wsi import extract_tiles from app.rules.engine import apply_clinical_rules from app.utils.config import CONFIG logger = logging.getLogger("cancerAI.predictor") _MC_DROPOUT_SAMPLES = 30 # Number of stochastic forward passes for uncertainty class CancerAIPredictor: def __init__(self): self._extractor: tf.keras.Model | None = None self._mil: tf.keras.Model | None = None self._treatment: tf.keras.Model | None = None self._scaler = None self._ready = False self._load_models() def _load_models(self): model_dir = CONFIG.model.dir mil_path = os.path.join(model_dir, "mil_production.keras") trt_path = os.path.join(model_dir, "treatment_production.keras") scaler_path = os.path.join(model_dir, "scaler.pkl") if all(os.path.exists(p) for p in [mil_path, trt_path, scaler_path]): self._extractor = build_tile_feature_extractor( input_shape=tuple(CONFIG.model.input_shape), fine_tune=False ) self._mil = tf.keras.models.load_model(mil_path) self._treatment = tf.keras.models.load_model(trt_path) self._scaler = joblib.load(scaler_path) self._ready = True logger.info("✅ Production models loaded successfully") else: logger.warning("⚠️ Production models not found — run training pipeline first") def is_ready(self) -> bool: return self._ready def _require_ready(self): if not self._ready: raise RuntimeError("Models are not loaded. Run the training pipeline first.") def _load_image(self, path: str) -> np.ndarray: """Load and normalise a standard image file.""" img = Image.open(path).resize( (CONFIG.model.input_shape[1], CONFIG.model.input_shape[0]) ).convert("RGB") return (np.array(img, dtype=np.float32) / 255.0) def _load_wsi_tiles(self, path: str) -> np.ndarray: """Load WSI and return stacked tile feature array for MIL.""" tiles = extract_tiles( path, tile_size=CONFIG.wsi.tile_size, level=CONFIG.wsi.level, min_tissue_ratio=CONFIG.wsi.min_tissue_ratio, max_tiles=CONFIG.mil.n_instances_max, target_shape=tuple(CONFIG.model.input_shape[:2]), ) if not tiles: raise ValueError("No tissue tiles extracted from WSI — check slide quality") arrays = np.stack([t.array for t in tiles]) return arrays # shape: (n_tiles, H, W, C) def _mc_predict_cancer(self, tile_features: np.ndarray) -> tuple[float, float]: """ Monte-Carlo Dropout cancer probability estimation. Args: tile_features: (n_tiles, feature_dim) Returns: (mean_prob, std_prob) — mean and standard deviation over MC samples """ bag = np.expand_dims(tile_features, axis=0) # (1, n_tiles, feature_dim) preds = [] for _ in range(_MC_DROPOUT_SAMPLES): # Dropout layers have training=True baked in — each call is stochastic p = float(self._mil(bag, training=False).numpy()[0, 0]) preds.append(p) return float(np.mean(preds)), float(np.std(preds)) def _mc_predict_treatment(self, genomic_scaled: np.ndarray) -> tuple[int, np.ndarray, float]: """ MC Dropout treatment recommendation. Returns: (treatment_idx, mean_probs, uncertainty_entropy) """ x = genomic_scaled.reshape(1, -1) all_probs = [] for _ in range(_MC_DROPOUT_SAMPLES): p = self._treatment(x, training=False).numpy()[0] all_probs.append(p) mean_probs = np.mean(all_probs, axis=0) # Predictive entropy as uncertainty measure entropy = float(-np.sum(mean_probs * np.log(mean_probs + 1e-9))) return int(np.argmax(mean_probs)), mean_probs, entropy def predict_image(self, image_path: str) -> dict: """Run cancer probability prediction on a single image or WSI.""" self._require_ready() is_wsi = image_path.lower().endswith((".svs", ".ndpi", ".scn", ".mrxs", ".tiff")) if is_wsi: tiles = self._load_wsi_tiles(image_path) tile_features = self._extractor.predict(tiles, batch_size=32, verbose=0) else: img = self._load_image(image_path) tile_features = self._extractor.predict(np.expand_dims(img, 0), verbose=0) mean_prob, std_prob = self._mc_predict_cancer(tile_features) return { "cancer_probability": round(mean_prob, 4), "uncertainty_std": round(std_prob, 4), "uncertainty_level": _uncertainty_label(std_prob), "n_tiles_analysed": len(tile_features), "is_wsi": is_wsi, } def predict_multimodal(self, image_path: str, genomic_payload: dict) -> dict: """Full multimodal inference: image + genomics → cancer prob + treatment.""" self._require_ready() # ── Image pathway ──────────────────────────────────────────────────── is_wsi = image_path.lower().endswith((".svs", ".ndpi", ".scn", ".mrxs", ".tiff")) if is_wsi: tiles = self._load_wsi_tiles(image_path) tile_features = self._extractor.predict(tiles, batch_size=32, verbose=0) else: img = self._load_image(image_path) tile_features = self._extractor.predict(np.expand_dims(img, 0), verbose=0) cancer_mean, cancer_std = self._mc_predict_cancer(tile_features) # ── Genomic pathway ────────────────────────────────────────────────── genomic_arr = np.array(genomic_payload["features"], dtype=np.float32) genomic_scaled = self._scaler.transform(genomic_arr.reshape(1, -1)) treatment_idx, treatment_probs, treatment_entropy = self._mc_predict_treatment(genomic_scaled[0]) # ── SHAP explanations ──────────────────────────────────────────────── shap_values = compute_shap_explanations( model=self._treatment, background_data=genomic_scaled, input_data=genomic_scaled, feature_names=genomic_payload.get("feature_names", []), ) # ── Clinical rules post-processing (no eval) ───────────────────────── metadata = { "age": genomic_payload.get("age", 0), "comorbidities": genomic_payload.get("comorbidities", []), } rule_result = apply_clinical_rules( cancer_prob=cancer_mean, treatment_idx=treatment_idx, metadata=metadata, treatments=CONFIG.treatments, rules=CONFIG.rules, ) return { "cancer_probability": round(cancer_mean, 4), "cancer_uncertainty_std": round(cancer_std, 4), "uncertainty_level": _uncertainty_label(cancer_std), "treatment_recommendation": rule_result.description, "treatment_idx": rule_result.treatment_idx, "treatment_probabilities": { CONFIG.treatments[i]: round(float(p), 4) for i, p in enumerate(treatment_probs) if i < len(CONFIG.treatments) }, "treatment_uncertainty_entropy": round(treatment_entropy, 4), "rules_applied": rule_result.rules_applied, "flagged_for_review": rule_result.flagged_for_review, "rules_reason": rule_result.reason, "shap_top_features": shap_values, "n_tiles_analysed": len(tile_features), } def _uncertainty_label(std: float) -> str: if std < 0.05: return "low" if std < 0.15: return "moderate" return "high" 

app/inference/explainer.py

""" SHAP explanations for the genomic treatment model. Uses KernelExplainer as a model-agnostic fallback (compatible with any Keras model regardless of architecture). DeepExplainer is faster but requires a specific graph structure — use it if your model allows. """ from __future__ import annotations import logging import numpy as np import shap import tensorflow as tf logger = logging.getLogger("cancerAI.explainer") def compute_shap_explanations( model: tf.keras.Model, background_data: np.ndarray, input_data: np.ndarray, feature_names: list[str], top_n: int = 10, ) -> list[dict]: """ Compute SHAP feature attributions for a genomic prediction. Uses KernelSHAP (model-agnostic). For performance in production, replace with shap.DeepExplainer if the model graph is compatible. Args: model: Trained Keras treatment model. background_data: Representative background dataset for SHAP baseline. input_data: Single sample to explain (shape: (1, n_features)). feature_names: List of genomic feature names. top_n: Return top N most influential features. Returns: List of dicts: [{"feature": str, "shap_value": float}, ...] """ try: # Summarise background with k-means to keep runtime manageable background_summary = shap.kmeans(background_data, min(10, len(background_data))) def model_predict(x): return model(x, training=False).numpy() explainer = shap.KernelExplainer(model_predict, background_summary) shap_values = explainer.shap_values(input_data, nsamples=100, silent=True) # shap_values shape: (n_classes, n_samples, n_features) or (n_samples, n_features) if isinstance(shap_values, list): # Multi-class: sum absolute values across classes combined = np.sum(np.abs(shap_values), axis=0)[0] else: combined = np.abs(shap_values[0]) top_indices = np.argsort(combined)[::-1][:top_n] names = feature_names if len(feature_names) == len(combined) else [f"feature_{i}" for i in range(len(combined))] return [ {"feature": names[i], "shap_value": round(float(combined[i]), 6)} for i in top_indices ] except Exception as e: logger.warning("SHAP explanation failed: %s — returning empty attributions", e) return []
