""" Fernet encryption for sensitive patient metadata and audit logs at rest. v9: Added field-level encryption helpers for structured dicts. PRODUCTION NOTE: Store ENCRYPTION_KEY in a secrets manager. """ from __future__ import annotations import os from cryptography.fernet import Fernet def _load_encryption_key() -> bytes: key = os.getenv("ENCRYPTION_KEY") if not key: raise RuntimeError( "ENCRYPTION_KEY environment variable is not set. " "CancerAI will not start without an encryption key. " "Generate one with: python -c \"from cryptography.fernet import Fernet; " "print(Fernet.generate_key().decode())\"" ) return key.encode() if isinstance(key, str) else key _cipher = Fernet(_load_encryption_key()) def encrypt(plaintext: str) -> str: """Encrypt a plaintext string, returning a URL-safe base64 token.""" return _cipher.encrypt(plaintext.encode()).decode() def decrypt(token: str) -> str: """Decrypt a Fernet token back to plaintext.""" return _cipher.decrypt(token.encode()).decode() def encrypt_fields(data: dict, fields: list[str]) -> dict: """ Encrypt specific fields in a dict in-place (returns new dict). Non-present fields are silently skipped. """ out = dict(data) for field in fields: if field in out and out[field] is not None: out[field] = encrypt(str(out[field])) return out def decrypt_fields(data: dict, fields: list[str]) -> dict: """Decrypt previously encrypted fields.""" out = dict(data) for field in fields: if field in out and out[field] is not None: out[field] = decrypt(str(out[field])) return out

""" SHA-256 hash verification for model files at startup. Prevents loading of tampered or corrupted model weights. """ from __future__ import annotations import hashlib import logging from pathlib import Path logger = logging.getLogger("cancerAI.security") def compute_sha256(path: str | Path) -> str: """Compute SHA-256 hash of a file, reading in chunks to handle large models.""" h = hashlib.sha256() with open(path, "rb") as f: for chunk in iter(lambda: f.read(1 << 20), b""): # 1 MB chunks h.update(chunk) return h.hexdigest() def verify_model_hash(path: str | Path, expected_hash: str) -> None: """ Verify a model file's SHA-256 hash against the expected value. Raises RuntimeError if the hash does not match or the file is missing. Args: path: Path to the .keras or .h5 model file. expected_hash: Expected SHA-256 hex digest (from config.yaml model_hashes). Usage: verify_model_hash("models/cancer_model.keras", config.model.model_hashes["cancer_model"]) """ path = Path(path) if not path.exists(): raise RuntimeError(f"Model file not found: {path}") if not expected_hash: logger.warning( "No expected hash configured for %s — skipping verification. " "Set model_hashes in config.yaml for production.", path.name ) return actual = compute_sha256(path) if actual != expected_hash: raise RuntimeError( f"Model hash mismatch for {path.name}!\n" f" Expected: {expected_hash}\n" f" Actual: {actual}\n" "The model file may be corrupted or tampered with. " "Do not proceed without resolving this." ) logger.info("✅ Model hash verified: %s", path.name)

""" Domain-specific pathology backbone adapter. v9 improvement: Replaces EfficientNetB3 (ImageNet) with support for UNI, CONCH, and PLIP — foundation models pre-trained on H&E histopathology. These substantially outperform ImageNet backbones on WSI tasks. Setup instructions: UNI: https://github.com/mahmoodlab/UNI (requires HuggingFace access token) CONCH: https://github.com/mahmoodlab/CONCH PLIP: https://github.com/PathologyFoundation/plip Set `backbone` and `backbone_weights_path` in config.yaml. If domain-specific weights are unavailable, falls back to EfficientNetB3. """ from __future__ import annotations import logging from pathlib import Path import numpy as np import tensorflow as tf from keras import layers, models logger = logging.getLogger("cancerAI.backbone") def build_backbone( backbone_name: str, weights_path: str | None, input_shape: tuple[int, int, int], freeze: bool = True, ) -> tf.keras.Model: """ Build and return a feature-extraction backbone. Args: backbone_name: One of "uni", "conch", "plip", "efficientnet_b3". weights_path: Local path to domain-specific weights directory. input_shape: (H, W, C) input shape for a single tile. freeze: Freeze backbone weights during initial training. Returns: A Keras model that maps (batch, H, W, C) → (batch, feature_dim). """ name = backbone_name.lower() if name in ("uni", "conch", "plip"): return _build_pathology_backbone(name, weights_path, input_shape, freeze) elif name == "efficientnet_b3": return _build_efficientnet(input_shape, freeze) else: logger.warning( "Unknown backbone '%s' — falling back to EfficientNetB3.", backbone_name ) return _build_efficientnet(input_shape, freeze) def _build_pathology_backbone( name: str, weights_path: str | None, input_shape: tuple[int, int, int], freeze: bool, ) -> tf.keras.Model: """ Adapter for UNI / CONCH / PLIP ViT-based pathology encoders. These models are Vision Transformers (ViT-L/16 or ViT-B/16) pre-trained on millions of H&E patches. Loading requires the weights to be available locally (downloaded separately due to institutional access requirements). This adapter wraps the encoder in a Keras-compatible interface. If weights are not found, raises RuntimeError with instructions. """ if not weights_path or not Path(weights_path).exists(): raise RuntimeError( f"Pathology backbone '{name}' requested but weights not found at: {weights_path}\n" f"Please download the weights from the respective repository and set " f"backbone_weights_path in config.yaml.\n" f"Alternatively, set backbone: 'efficientnet_b3' to use the ImageNet fallback." ) logger.info("Loading %s pathology backbone from %s", name.upper(), weights_path) # ── Attempt to load via SavedModel format (standard export from UNI/CONCH/PLIP) ── try: encoder = tf.saved_model.load(weights_path) logger.info("%s backbone loaded successfully.", name.upper()) except Exception as e: raise RuntimeError( f"Failed to load {name.upper()} backbone from {weights_path}: {e}" ) from e # ── Wrap in a Keras layer for seamless integration ──────────────────────── inputs = layers.Input(shape=input_shape, name="tile_input") # Pathology encoders typically expect [0, 1] normalised float32 input # and return CLS token embeddings of shape (batch, embed_dim) features = layers.Lambda( lambda x: encoder(x, training=False), name=f"{name}_encoder", )(inputs) if freeze: # Mark the Lambda as non-trainable proxy; actual freezing # is enforced by not including encoder variables in optimizer scope. logger.info("Backbone frozen for initial training.") backbone_model = models.Model(inputs=inputs, outputs=features, name=f"{name}_backbone") return backbone_model def _build_efficientnet( input_shape: tuple[int, int, int], freeze: bool, ) -> tf.keras.Model: """EfficientNetB3 fallback (ImageNet weights).""" logger.warning( "Using EfficientNetB3 (ImageNet) backbone. " "For histopathology tasks, consider domain-specific backbones: UNI, CONCH, or PLIP." ) base = tf.keras.applications.EfficientNetB3( include_top=False, weights="imagenet", input_shape=input_shape, pooling="avg", ) base.trainable = not freeze return base

""" Tile-level CNN model using a configurable domain-specific backbone. v9: Backbone is now swappable via config (UNI / CONCH / PLIP / EfficientNetB3). """ from __future__ import annotations import logging import tensorflow as tf from keras import layers, models, regularizers from app.models.backbone import build_backbone from app.utils.config import CONFIG logger = logging.getLogger("cancerAI.cnn") def build_tile_encoder() -> tf.keras.Model: """ Build the tile-level feature encoder. Backbone outputs a feature vector per tile; a projection head maps to a fixed embedding dimension for MIL aggregation. Returns: Keras model: (batch, H, W, C) → (batch, projection_dim) """ cfg = CONFIG.model input_shape = tuple(cfg.input_shape) backbone = build_backbone( backbone_name=cfg.backbone, weights_path=cfg.backbone_weights_path, input_shape=input_shape, freeze=True, ) # ── Projection head ──────────────────────────────────────────────────────── inputs = layers.Input(shape=input_shape, name="tile_input") features = backbone(inputs) # Project to consistent embedding dim regardless of backbone output size x = layers.Dense( 512, activation="relu", kernel_regularizer=regularizers.l2(cfg.l2_reg), name="projection_fc", )(features) x = layers.Dropout(cfg.dropout, name="projection_dropout")(x) x = layers.Dense(256, activation="relu", name="embedding_fc")(x) encoder = models.Model(inputs=inputs, outputs=x, name="tile_encoder") logger.info( "Tile encoder built: backbone=%s → projection_dim=256", cfg.backbone ) return encoder 

app/models/mil.py

""" Attention-based Multiple Instance Learning (MIL) aggregator. v9 fix: training flag is now correctly propagated so MC Dropout is stochastic when training=True during uncertainty estimation. """ from __future__ import annotations import logging import tensorflow as tf from keras import layers, models, regularizers from app.utils.config import CONFIG logger = logging.getLogger("cancerAI.mil") class AttentionMIL(tf.keras.Model): """ Attention-based MIL aggregator (Ilse et al., 2018). Takes a bag of tile embeddings and produces a slide-level prediction via learned attention weights. Critical v9 fix: All Dropout layers are called with the `training` argument explicitly passed through from the model's call() signature. This ensures: - training=False during standard inference (dropout OFF) - training=True during MC Dropout uncertainty sampling (dropout ON) Previously, training=False was hardcoded in _mc_predict_cancer, making MC uncertainty estimates meaningless. """ def __init__( self, embed_dim: int = 256, attention_dim: int = 128, n_classes: int = 1, dropout_rate: float = 0.3, l2_reg: float = 1e-4, **kwargs, ): super().__init__(**kwargs) self.embed_dim = embed_dim self.attention_dim = attention_dim self.n_classes = n_classes self.dropout_rate = dropout_rate reg = regularizers.l2(l2_reg) # Attention network: V and U branches (gated attention) self.attn_V = layers.Dense(attention_dim, activation="tanh", kernel_regularizer=reg) self.attn_U = layers.Dense(attention_dim, activation="sigmoid", kernel_regularizer=reg) self.attn_w = layers.Dense(1, use_bias=False) # Classifier head self.dropout = layers.Dropout(dropout_rate) self.classifier = layers.Dense(n_classes, activation="sigmoid", kernel_regularizer=reg) def call(self, inputs: tf.Tensor, training: bool = False) -> dict: """ Args: inputs: Tile embeddings, shape (batch, n_tiles, embed_dim). training: If True, Dropout is active (used for MC Dropout sampling). Returns: dict with keys: "logits": (batch, n_classes) slide-level prediction "attention": (batch, n_tiles, 1) per-tile attention weights """ # Gated attention scores v = self.attn_V(inputs) # (B, T, attention_dim) u = self.attn_U(inputs) # (B, T, attention_dim) e = self.attn_w(v * u) # (B, T, 1) element-wise gate alpha = tf.nn.softmax(e, axis=1) # (B, T, 1) normalised attention # Weighted aggregation z = tf.reduce_sum(alpha * inputs, axis=1) # (B, embed_dim) # Classifier with dropout — training flag properly forwarded z = self.dropout(z, training=training) logits = self.classifier(z) # (B, n_classes) return {"logits": logits, "attention": alpha} def get_config(self) -> dict: return { "embed_dim": self.embed_dim, "attention_dim": self.attention_dim, "n_classes": self.n_classes, "dropout_rate": self.dropout_rate, } def build_mil_model(tile_encoder: tf.keras.Model) -> tf.keras.Model: """ Compose the full MIL model: tile encoder + attention aggregator. Used during bag-level training. Input: (batch, n_tiles, H, W, C) Output: {"logits": (batch, 1), "attention": (batch, n_tiles, 1)} """ cfg = CONFIG.model n_tiles = CONFIG.mil.n_instances_max h, w, c = cfg.input_shape bag_input = layers.Input(shape=(None, h, w, c), name="bag_input") # variable n_tiles # Encode each tile: apply encoder across the tile dimension tile_features = layers.TimeDistributed(tile_encoder, name="tile_encoder_td")(bag_input) # tile_features: (B, n_tiles, embed_dim) mil = AttentionMIL( attention_dim=CONFIG.mil.attention_dim, dropout_rate=cfg.dropout, l2_reg=cfg.l2_reg, name="attention_mil", ) outputs = mil(tile_features) model = models.Model(inputs=bag_input, outputs=outputs, name="cancerAI_mil") return model

""" Whole Slide Image tiling pipeline. v9: Parallelised tile extraction using ThreadPoolExecutor. Sequential loop replaced with concurrent workers (configurable via config.yaml wsi.max_workers). """ from __future__ import annotations import logging from concurrent.futures import ThreadPoolExecutor, as_completed from dataclasses import dataclass import cv2 import numpy as np import openslide from app.utils.config import CONFIG logger = logging.getLogger("cancerAI.wsi") @dataclass class Tile: array: np.ndarray # float32, shape (H, W, 3), normalised 0–1 x: int y: int tissue_ratio: float def _process_tile( slide_path: str, x: int, y: int, tile_size: int, level: int, target_shape: tuple[int, int], saturation_threshold: int, value_threshold: int, min_tissue_ratio: float, ) -> Tile | None: """Extract and filter a single tile. Returns None if below tissue threshold.""" with openslide.OpenSlide(slide_path) as slide: tile_img = slide.read_region((x, y), level, (tile_size, tile_size)) tile_rgb = np.array(tile_img.convert("RGB"), dtype=np.uint8) hsv = cv2.cvtColor(tile_rgb, cv2.COLOR_RGB2HSV) tissue_mask = ( (hsv[:, :, 1] > saturation_threshold) & (hsv[:, :, 2] > value_threshold) ) tissue_ratio = float(np.mean(tissue_mask)) if tissue_ratio < min_tissue_ratio: return None resized = cv2.resize(tile_rgb, (target_shape[1], target_shape[0])) return Tile( array=(resized.astype(np.float32) / 255.0), x=x, y=y, tissue_ratio=tissue_ratio, ) def extract_tiles( slide_path: str, tile_size: int | None = None, level: int | None = None, min_tissue_ratio: float | None = None, max_tiles: int = 200, target_shape: tuple[int, int] = (300, 300), saturation_threshold: int = 30, value_threshold: int = 50, max_workers: int | None = None, ) -> list[Tile]: """ Extract tissue-containing tiles from a WSI using parallel workers. Args: slide_path: Path to .svs / .tiff / .ndpi slide. tile_size: Tile edge length (default from config). level: OpenSlide pyramid level (default from config). min_tissue_ratio: Minimum tissue fraction (default from config). max_tiles: Hard cap to prevent OOM. target_shape: Resize target (H, W). saturation_threshold: HSV saturation threshold for tissue detection. value_threshold: HSV value threshold for tissue detection. max_workers: Parallel threads (default from config wsi.max_workers). Returns: List of Tile objects sorted by tissue_ratio descending. """ cfg = CONFIG.wsi tile_size = tile_size or cfg.tile_size level = level if level is not None else cfg.level min_tissue_ratio = min_tissue_ratio or cfg.min_tissue_ratio max_workers = max_workers or getattr(cfg, "max_workers", 4) with openslide.OpenSlide(slide_path) as slide: width, height = slide.level_dimensions[level] logger.info("WSI %s: %d × %d px at level %d", slide_path, width, height, level) # Build candidate (x, y) coordinates coords = [ (x, y) for y in range(0, height, tile_size) for x in range(0, width, tile_size) ] tiles: list[Tile] = [] with ThreadPoolExecutor(max_workers=max_workers) as executor: futures = { executor.submit( _process_tile, slide_path, x, y, tile_size, level, target_shape, saturation_threshold, value_threshold, min_tissue_ratio, ): (x, y) for x, y in coords } for future in as_completed(futures): if len(tiles) >= max_tiles: # Cancel remaining futures once cap reached for f in futures: f.cancel() break result = future.result() if result is not None: tiles.append(result) if len(tiles) >= max_tiles: break tiles.sort(key=lambda t: t.tissue_ratio, reverse=True) tiles = tiles[:max_tiles] logger.info("Extracted %d high-quality tiles (parallel workers=%d)", len(tiles), max_workers) return tiles

""" Bag-level MIL K-Fold training pipeline. v9: Proper bag-level training with variable n_tiles per slide. Previously each slide was treated as a single tile (n_tiles=1), which did not exercise the MIL attention mechanism during training. """ from __future__ import annotations import logging from pathlib import Path from typing import Generator import numpy as np import tensorflow as tf from sklearn.model_selection import StratifiedKFold from sklearn.preprocessing import StandardScaler from app.models.cnn import build_tile_encoder from app.models.mil import build_mil_model from app.pipelines.wsi import extract_tiles from app.utils.config import CONFIG logger = logging.getLogger("cancerAI.training") def slide_to_bag( slide_path: str, min_tiles: int, max_tiles: int, target_shape: tuple[int, int], ) -> np.ndarray | None: """ Extract tiles from a slide and return a bag array of shape (n_tiles, H, W, C). Each "sample" is a slide (bag of tiles). The model sees variable numbers of tiles per slide per epoch, giving the attention mechanism meaningful input to learn from. Args: slide_paths: List of paths to WSI files. labels: Binary labels (0=benign, 1=malignant), one per slide. save_dir: Directory to save per-fold model weights. Returns: List of per-fold metric dicts. """ cfg = CONFIG save_path = Path(save_dir) save_path.mkdir(parents=True, exist_ok=True) labels_arr = np.array(labels) skf = StratifiedKFold( n_splits=cfg.training.k_folds, shuffle=True, random_state=42 ) fold_results = [] target_shape = tuple(cfg.model.input_shape[:2]) # (H, W) for fold, (train_idx, val_idx) in enumerate(skf.split(slide_paths, labels_arr)): logger.info("── Fold %d/%d ──", fold + 1, cfg.training.k_folds) train_paths = [slide_paths[i] for i in train_idx] train_labels = labels_arr[train_idx].tolist() val_paths = [slide_paths[i] for i in val_idx] val_labels = labels_arr[val_idx].tolist() # Build fresh model per fold tile_encoder = build_tile_encoder() model = build_mil_model(tile_encoder) model.compile( optimizer=tf.keras.optimizers.Adam(1e-4), loss={"logits": "binary_crossentropy"}, metrics={"logits": ["AUC", "accuracy"]}, ) best_val_auc = 0.0 patience_counter = 0 for epoch in range(cfg.training.epochs): train_loss_accum = [] # Add batch dimension: (1, n_tiles, H, W, C) bag_batch = np.expand_dims(bag, axis=0) label_batch = np.array([[label]], dtype=np.float32) with tf.GradientTape() as tape: outputs = model(bag_batch, training=True) loss = tf.keras.losses.binary_crossentropy( label_batch, outputs["logits"] ) grads = tape.gradient(loss, model.trainable_variables) model.optimizer.apply_gradients( zip(grads, model.trainable_variables) ) train_loss_accum.append(float(loss)) # app/inference/explainer.py

""" Explainability module. v9: Replaced slow KernelExplainer with DeepExplainer (primary) and IntegratedGradients via tf-keras-vis (fallback). Performance comparison (approximate): KernelExplainer: O(n_samples * n_features) — very slow, model-agnostic DeepExplainer: ~10-50x faster, uses backprop through model IntegratedGradients: Fast, theoretically grounded, works on any differentiable model """ from __future__ import annotations import logging import numpy as np import tensorflow as tf logger = logging.getLogger("cancerAI.explainer") def explain_prediction( model: tf.keras.Model, bag: np.ndarray, attention_weights: list[float], method: str = "integrated_gradients", n_steps: int = 50, ) -> dict: """ Generate tile-level feature attribution for a MIL prediction. Args: model: The loaded CancerAI MIL model. bag: (1, n_tiles, H, W, C) tile bag. attention_weights: Per-tile attention weights from the MIL head. method: "integrated_gradients" (default) or "deep_explainer". n_steps: Integration steps for IG (higher = more accurate, slower). Returns: dict with: "method": Attribution method used. "top_tiles": Indices of top-3 most attended tiles. "tile_attributions": Per-tile importance scores (normalised). """ try: if method == "integrated_gradients": return _integrated_gradients(model, bag, attention_weights, n_steps) elif method == "deep_explainer": return _deep_explainer(model, bag, attention_weights) else: raise ValueError(f"Unknown explainability method: {method}") except Exception as e: logger.warning("Explainability failed (%s): %s — returning attention-only summary.", method, e) return _attention_only_summary(attention_weights) def _integrated_gradients( model: tf.keras.Model, bag: np.ndarray, attention_weights: list[float], n_steps: int, ) -> dict: """ Compute Integrated Gradients for each tile in the bag. Baseline: all-zero bag (represents absence of tissue signal). IG formula: IG(x) = (x - x') * integral[dF/dx from x' to x] Approximated via Riemann summation over n_steps. """ bag_tensor = tf.constant(bag, dtype=tf.float32) baseline = tf.zeros_like(bag_tensor) # Interpolate between baseline and input alphas = tf.linspace(0.0, 1.0, n_steps + 1) interpolated = tf.stack( [baseline + a * (bag_tensor - baseline) for a in alphas], axis=0 ) # (n_steps+1, 1, n_tiles, H, W, C) # Compute gradients at each step with tf.GradientTape() as tape: tape.watch(interpolated) # Reshape for batched forward pass flat = tf.reshape(interpolated, [-1] + list(bag_tensor.shape[1:])) outputs = model(flat, training=False) probs = outputs["logits"][:, 0] grads = tape.gradient(probs, interpolated) # same shape as interpolated # Trapezoidal integration (average of adjacent steps) avg_grads = tf.reduce_mean((grads[:-1] + grads[1:]) / 2.0, axis=0) integrated = (bag_tensor - baseline) * avg_grads # (1, n_tiles, H, W, C) # Tile-level importance: mean absolute attribution per tile tile_attrs = tf.reduce_mean(tf.abs(integrated[0]), axis=[1, 2, 3]).numpy() tile_attrs = tile_attrs / (tile_attrs.sum() + 1e-9) # normalise top_tiles = np.argsort(tile_attrs)[::-1][:3].tolist() return { "method": "integrated_gradients", "top_tiles": top_tiles, "tile_attributions": tile_attrs.tolist(), } def _deep_explainer( model: tf.keras.Model, bag: np.ndarray, attention_weights: list[float], ) -> dict: """ SHAP DeepExplainer for faster attribution (requires shap library). Uses a zero background for consistency with IG baseline choice. """ import shap background = np.zeros_like(bag) explainer = shap.DeepExplainer(model, background) try: shap_values = explainer.shap_values(bag) # shap_values: list of arrays for each output; take first output sv = np.abs(shap_values[0]) if isinstance(shap_values, list) else np.abs(shap_values) tile_attrs = sv[0].mean(axis=(1, 2, 3)) # (n_tiles,) tile_attrs = tile_attrs / (tile_attrs.sum() + 1e-9) top_tiles = np.argsort(tile_attrs)[::-1][:3].tolist() return { "method": "deep_explainer", "top_tiles": top_tiles, "tile_attributions": tile_attrs.tolist(), } except Exception as e: logger.warning("DeepExplainer failed: %s — falling back to IntegratedGradients.", e) return _integrated_gradients(model, bag, attention_weights, n_steps=25) def _attention_only_summary(attention_weights: list[float]) -> dict: """Fallback: use MIL attention weights as proxy for tile importance.""" weights = np.array(attention_weights) top_tiles = np.argsort(weights)[::-1][:3].tolist() return { "method": "attention_weights_only", "top_tiles": top_tiles, "tile_attributions": weights.tolist(), }

If multiple rules match, only the highest-priority one is applied. `flag_for_review` is OR'd across all matching rules (any flag = flagged). """ context = { "cancer_prob": cancer_prob, "treatment_idx": treatment_idx, "age": metadata.get("age", 0), "comorbidities": metadata.get("comorbidities", []), # Extended v9 fields "stage": metadata.get("stage", None), "ecog_status": metadata.get("ecog_status", None), "pdl1_expression": metadata.get("pdl1_expression", None), "msi_status": metadata.get("msi_status", None), "brca_mutation": metadata.get("brca_mutation", None), } # Sort rules by priority (ascending = highest priority first) sorted_rules = sorted(rules, key=lambda r: r.get("priority", 100)) matched_rules: list[dict] = [] flagged = False for rule in sorted_rules: rule_id = rule.get("id", "unnamed") try: if _evaluate_condition(rule["condition"], context): matched_rules.append(rule) if rule.get("action", {}).get("flag_for_review"): flagged = True except Exception as e: logger.error("Rule '%s' evaluation error: %s — skipped", rule_id, e) if not matched_rules: desc = treatments[treatment_idx] if 0 <= treatment_idx < len(treatments) else "Unknown" return RuleResult( treatment_idx=treatment_idx, description=desc, reason="Model recommendation (no rules matched)", flagged_for_review=False, rules_applied=[], ) # Apply highest-priority matching rule (first in sorted list) winning_rule = matched_rules[0] action = winning_rule.get("action", {}) final_idx = action.get("set_treatment", treatment_idx) reason = action.get("reason", "Clinical rule applied") applied_ids = [r.get("id", "unnamed") for r in matched_rules] logger.info( "Rules matched: %s. Winning rule: '%s' → treatment %d → %d", applied_ids, winning_rule.get("id"), treatment_idx, final_idx, ) desc = treatments[final_idx] if 0 <= final_idx < len(treatments) else "Unknown" return RuleResult( treatment_idx=final_idx, description=desc, reason=reason, flagged_for_review=flagged, rules_applied=applied_ids, )

""" Distribution drift detection for image features and genomic inputs. v9: Uses Kolmogorov-Smirnov test to detect covariate shift between the training distribution and live inference distribution. Drift in input distributions can silently degrade model performance — particularly important in clinical AI where data sources and staining protocols may change over time. """ from __future__ import annotations import logging from collections import deque import numpy as np from scipy import stats from app.utils.config import CONFIG logger = logging.getLogger("cancerAI.drift") class DriftDetector: """ Maintains a sliding window of recent predictions and compares the feature distribution to a stored baseline using the KS test. Usage: detector = DriftDetector() detector.set_baseline(training_cancer_probs) detector.update(new_cancer_prob) if detector.is_drifted(): alert(...) """ def __init__(self, window_size: int | None = None, ks_threshold: float | None = None): cfg = CONFIG.monitoring self._window_size = window_size or cfg.drift_window_size self._ks_threshold = ks_threshold or cfg.drift_ks_threshold self._window: deque[float] = deque(maxlen=self._window_size) self._baseline: np.ndarray | None = None self._drift_detected = False def set_baseline(self, values: list[float] | np.ndarray) -> None: """Set the reference distribution (e.g., from training set predictions).""" self._baseline = np.array(values, dtype=np.float64) logger.info( "Drift detector baseline set: n=%d, mean=%.4f, std=%.4f", len(self._baseline), self._baseline.mean(), self._baseline.std(), ) def update(self, value: float) -> None: """Add a new observation and check for drift if window is full.""" self._window.append(value) if len(self._window) >= self._window_size and self._baseline is not None: self._check_drift() def _check_drift(self) -> None: """Run KS test between current window and baseline.""" window_arr = np.array(self._window) ks_stat, p_value = stats.ks_2samp(self._baseline, window_arr) if p_value < self._ks_threshold: self._drift_detected = True logger.warning( "⚠️ Distribution drift detected! KS statistic=%.4f, p-value=%.6f " "(threshold=%.4f). Input distribution may have shifted from training.", ks_stat, p_value, self._ks_threshold, ) else: self._drift_detected = False def is_drifted(self) -> bool: return self._drift_detected def reset(self) -> None: self._window.clear() self._drift_detected = False # Module-level singletons — one detector per modality cancer_prob_drift = DriftDetector() genomic_drift = DriftDetector()

""" ONNX export script for CancerAI v9. Exports the tile encoder CNN to ONNX for high-throughput GPU/TPU serving via TF Serving, Triton, or ONNX Runtime. Usage: python scripts/export_onnx.py --model models/cancer_model.keras --output models/cancer_model.onnx Notes: - MIL aggregation (variable-length bag) is NOT exported to ONNX as ONNX has limited support for dynamic sequence models. Export the tile encoder only. - For production throughput, tile encoding can be batched and the ONNX model run in parallel; MIL aggregation runs as a lightweight NumPy post-processing step. """ from __future__ import annotations import argparse import logging from pathlib import Path import numpy as np import tensorflow as tf logging.basicConfig(level=logging.INFO) logger = logging.getLogger("cancerAI.export_onnx") def export_to_onnx(model_path: str, output_path: str, input_shape: tuple) -> None: import tf2onnx import onnx logger.info("Loading model from %s ...", model_path) model = tf.keras.models.load_model(model_path) # Export tile encoder only (static input shape) spec = (tf.TensorSpec(input_shape, tf.float32, name="tile_input"),) output_path_resolved = str(Path(output_path)) logger.info("Converting to ONNX (opset 17)...") model_proto, _ = tf2onnx.convert.from_keras( model, input_signature=spec, opset=17, output_path=output_path_resolved ) # Validate exported model onnx.checker.check_model(output_path_resolved) logger.info("✅ ONNX model saved and validated: %s", output_path_resolved) # Quick inference check import onnxruntime as ort sess = ort.InferenceSession(output_path_resolved) dummy = np.random.rand(*input_shape[1:]).astype(np.float32)[np.newaxis] result = sess.run(None, {"tile_input": dummy}) logger.info("ONNX inference check passed. Output shape: %s", result[0].shape) if __name__ == "__main__": parser = argparse.ArgumentParser(description="Export CancerAI tile encoder to ONNX") parser.add_argument("--model", default="models/cancer_model.keras") parser.add_argument("--output", default="models/cancer_model.onnx") parser.add_argument("--height", type=int, default=300) parser.add_argument("--width", type=int, default=300) args = parser.parse_args() export_to_onnx( model_path=args.model, output_path=args.output, input_shape=(None, args.height, args.width, 3), )
